27  /**
28   * <p>Build a TCS fragment</p>
29   * 
30   * <p>The packaging rules are as follows :-</p>
31   * <ol>
32   * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
33   *      <ul>
34   *              <li>Archiver-Version: Plexus Archiver</li>
35   *              <li>Built-By: build user</li>
36   *              <li>Build-Jdk: jdk version</li>
37   *              <li>Package-Title: project groupId and name</li>
38   *              <li>Package-Version: project version and buildNumber</li>
39   *              <li>Package-Vendor: project organization name (if set)</li>
40   *              <li>TIBCO-EP-Build-Product-Version: product version</li>
41   *      </ul>
42   * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
43   * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
44   * <li>The project's files are copied into /</li>
45   * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
46   * <li>The project's resources files are copied into /</li>
47   * </ol>
48   * 
49   * <p>The plexus archiver is used to create the archive via the maven assembly
50   * plugin.</p>
51   * 
52   * <p>Native archives (nar) are included - in this case a mapping is included
53   * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
54   * values.</p>
55   * 
56   * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-tcs-fragment.zip</p>
57   * 
58   */
59  @Mojo(name = "package-tcs-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true)
60  public class PackageTCSFragmentMojo extends BasePackageMojo {
61  
62      /**
63       * <p>Additional resources directory for HOCON configurations</p>
64       * 
65       * <p>This is added to the list of resource directories</p>
66       * 
67       * <p>Example use in pom.xml:</p>
68       * <img src="uml/package-configurationDirectory.svg" alt="pom">
69       * 
70       * @since 1.0.0
71       */
72      @Parameter( defaultValue = "${project.basedir}/src/main/configurations", required = true )
73      File configurationDirectory;
74  
75      /**
76       * <p>Additional resources directory for test HOCON configurations</p>
77       * 
78       * <p>This is added to the list of test resource directories</p>
79       * 
80       * <p>Example use in pom.xml:</p>
81       * <img src="uml/package-testConfigurationDirectory.svg" alt="pom">
82       * 
83       * @since 1.0.0
84       */
85      @Parameter( defaultValue = "${project.basedir}/src/test/configurations", required = true )
86      File testConfigurationDirectory;
87  
88  
89      public void execute() throws MojoExecutionException {
90          getLog().debug( "Creating TCS fragment" );
91  
92          prechecks();
93  
94          Assembly assembly = createAssembly();
95  
96          // add runtime dependencies
97          //
98          // we add them one-by-one instead of using a dependencyset so we can
99          // set the classpath in the manifest file
100         //
101         String classPath = "";
102         String nativeClassPath = "";
103         String fragmentList = "";
104 
105         List<FileItem> deps = new ArrayList<FileItem>();
106         for (Artifact dependency : getCompileProjectDependencies()) {
107             String type = dependency.getType();
108             String destName = dependency.getGroupId()+"-"+dependency.getArtifactId()+"-"+dependency.getBaseVersion();
109 
110             if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
111                 destName += "-"+type+".zip";
112             } else {
113                 if (dependency.hasClassifier()) {
114                     destName += "-"+dependency.getClassifier();
115                 }
116                 destName += "."+dependency.getType();
117             }
118 
119             FileItem dependencyFile = new FileItem();
120             dependencyFile.setOutputDirectory("");
121             dependencyFile.setSource(getArtifactPath(dependency));
122             dependencyFile.setDestName(destName);
123             deps.add(dependencyFile);
124 
125             if (dependency.getType().equals("nar")) {
126                 nativeClassPath += destName+" ";
127             } else if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
128                 fragmentList += destName+" ";
129             } else {
130                 classPath += destName+" ";
131             }
132         }
133 
134         // build a manifest file and add as first file to the zip
135         //
136         Map<String, String> extras = new HashMap<String, String>();
137         extras.put("TIBCO-EP-Fragment-Format-Version", "2");
138         if (!classPath.isEmpty()) {
139             extras.put("Class-Path", classPath);
140         }
141         if (!nativeClassPath.isEmpty()) {
142             extras.put("TIBCO-EP-Native-Class-Path", nativeClassPath);
143         }
144         if (!fragmentList.isEmpty()) {
145             extras.put("TIBCO-EP-Fragment-List", fragmentList);
146         }            
147         File manifest = packageManifest(assembly, null, extras);
148 
149         for (FileItem dependencyFile : deps) {
150             assembly.addFile(dependencyFile);
151         }
152 
153         // add output directory
154         //
155         // this should include resources ( already copied here ) and
156         // eventflow files
157         //
158         FileSet fileSet = new FileSet();
159         fileSet.setDirectory(project.getBuild().getOutputDirectory());
160         fileSet.setOutputDirectory("");
161         assembly.addFileSet(fileSet);
162 
163         // write assembly
164         //
165         writeAssembly(assembly);
166         if (!manifest.delete()) {
167             getLog().debug("Unable to delete "+manifest.getAbsolutePath());
168         }
169     }
170 }
