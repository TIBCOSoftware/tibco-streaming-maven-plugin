53  /**
54   * <p>
55   * A Reporter that reports results via a socket connection to Eclipse (or
56   * StreamBase Studio) and its JUnit view, using a syntax and protocol specific
57   * to that view.
58   * </p>
59   * <p>
60   * For the sake of development-level testing, the constructor accepts two
61   * debug-output booleans. When true, "traceCalls" prints to stdout whenever a
62   * Reporter interface call is made. When true, "traceSocketMessages" prints to
63   * stdout a copy of the messages sent to the JUnit client via the Socket
64   * connection.
65   * </p>
66   */
67  public class EclipseJUnitViewSocketReporter extends AbstractReporter {
68

75      protected final boolean fTraceCalls;
76  
77      /** Is tracing of socket messages enabled? */
78      protected final boolean fTraceSocketMessages;

81      protected Socket fClientSocket;

103     /**
104      * Constructor
105      * 
106      * @param testResultsPort test results port number
107      * @param traceCalls true to trace calls
108      * @param traceSocketMessages true to trace socket messages
109      * @param testClasses test case classes
110      */
111     public EclipseJUnitViewSocketReporter(Integer testResultsPort, Boolean traceCalls, Boolean traceSocketMessages,
112             String[] testClasses) {
113         super(false);
114         fPort = testResultsPort;
115         fTraceCalls = traceCalls;
116         fTraceSocketMessages = traceSocketMessages;
117         fTestClasses = Arrays.copyOf(testClasses, testClasses.length);
118 
119         if (fTraceCalls) {
120             System.out.println("new SocketReporter(port = " + fPort + ", traceCalls = " + traceCalls
121                     + ", traceSocketMessages = " + traceSocketMessages + ")");
122         }
123         connectToClient();
124     }

128     /** Forms the Socket connection to the JUnit client in Eclipse/Studio */
129     protected void connectToClient() {
130         if (fTraceCalls) {
131             System.out.println("RemoteTestRunner: trying to connect to port " + fPort); //$NON-NLS-1$ //$NON-NLS-2$
132         }
133         Exception exception = null;
134         // Try forming the Socket connection up to 40 times, waiting a second between
135         // each try:
136         for (int i = 1; i < 40; i++) {
137             try {
138                 fClientSocket = new Socket((String) null, fPort);
139                 fWriter = new PrintWriter(
140                         new BufferedWriter(new OutputStreamWriter(fClientSocket.getOutputStream(), StandardCharsets.UTF_8)),
141                         false);
142 
143                 // See SB-43820:
144                 // If we support re-running tests in the future, then this Reader will be useful for getting re-run requests from the client.
145                 // See "org.eclipse.jdt.internal.junit.runner.RemoteTestRunner" for more details.
146                 /*
147                 try {
148                     fReader= new BufferedReader(new InputStreamReader(fClientSocket.getInputStream(), "UTF-8")); //$NON-NLS-1$
149                 } catch (UnsupportedEncodingException e1) {
150                     fReader= new BufferedReader(new InputStreamReader(fClientSocket.getInputStream()));
151                 }
152                 fReaderThread= new ReaderThread();
153                 fReaderThread.start();
154                 */
155                 if (fTraceCalls) {
156                     System.out.println("RemoteTestRunner: successfully connected to port " + fPort); //$NON-NLS-1$
157                 }
158                 return; // We didn't throw an Exception, so: success !!
159             } catch (IOException e) {
160                 exception = e;
161             }
162             try {
163                 Thread.sleep(1000);
164             } catch (InterruptedException e) {
165             }
166         }
167         if (fTraceCalls) {
168             System.out.println("RemoteTestRunner: failed to connect to port " + fPort); //$NON-NLS-1$
169         }
170         runFailed("Could not connect to port " + fPort, exception);
171     }

197         // See "org.eclipse.jdt.internal.junit.runner.RemoteTestRunner" for more details.

201         //        // on a lock held by the BufferedReader
202         //        // fix for bug: 38955

204         //    }

227     @Override
228     public void runStarting(int testCount) {
229         super.runStarting(testCount);
230         if (fTraceCalls) {
231             System.out.println("runStarting(" + testCount + ")"); //$NON-NLS-1$ //$NON-NLS-2$
232         }
233         fTestStartTime = System.currentTimeMillis();
234 
235         // The JUnit view wants a count of the number of test methods, but unfortunately
236         // "testCount" is the number of test files. So we need to count the number of
237         // methods,
238         // using the test class list given to us in the constructor:
239         int numberOfTestMethods = countNumberOfTestMethods();
240         sendMessage(JUnitViewMessageIds.TEST_RUN_START + numberOfTestMethods + " " + "v2"); //$NON-NLS-1$ //$NON-NLS-2$
241     }
242 
243     protected int countNumberOfTestMethods() {
244         int result = 0;
245         for (String testClass : fTestClasses) {
246             result += countNumberOfTestMethodsIn(testClass);
247         }
248         return result;
249     }
250 
251     protected int countNumberOfTestMethodsIn(String testClass) {
252         List<Method> methods;
253         try {
254             methods = testMethodsIn(testClass);
255         } catch (ClassNotFoundException e) {
256             return 0;
257         }
258         return methods.size();
259     }

298     @Override
299     public void testSetStarting(ReportEntry entry) throws ReporterException {
300         super.testSetStarting(entry);
301         if (fTraceCalls) {
302             System.out.println("testSetStarting(ReportEntry entry)"); //$NON-NLS-1$
303             printReportEntry(entry);
304         }
305 
306         // The JUnit protocol wants a TEST_TREE line for the "test-set" (i.e. entire
307         // Java file)
308         // as well as a TEST_TREE line for each test method.
309         //
310         // Produce the line for the Java file first:
311         boolean isSuite = true;
312         List<Method> methods;
313         try {
314             methods = testMethodsIn(entry.getName());
315         } catch (ClassNotFoundException e) {
316             throw new ReporterException("Class not found: " + entry.getName(), e);
317         }
318         int testCount = methods.size();
319         String treeEntry = getTestIdAndName(entry) + ',' + isSuite + ',' + testCount;
320         sendMessage(JUnitViewMessageIds.TEST_TREE + treeEntry);
321 
322         // Now emit a TEST_TREE line for each test method:
323         String fileName = entry.getName();
324         for (Method methodName : methods) {
325             // E.g. "2,test1(com.tibco.junit_step_through.TestCase),false,1"
326             isSuite = false;
327             String entryName = methodName.getName() + "(" + fileName + ")";
328             int id = getTestId(entryName);
329             String testMethodEntry = "" + id + "," + entryName + "," + isSuite + ",1";
330             sendMessage(JUnitViewMessageIds.TEST_TREE + testMethodEntry);
331         }
332     }
333 
334     protected List<Method> testMethodsIn(String testName) throws ClassNotFoundException {
335         Class<?> testClass = Class.forName(testName);
336         ArrayList<Method> result = new ArrayList<>();
337         Method[] methods = testClass.getMethods();
338         for (Method m : methods) {
339             if (isTestMethod(m)) {
340                 result.add(m);
341             }
342         }
343         return result;
344     }
345 
346     protected boolean isTestMethod(Method m) {
347         Annotation testAnnotation = m.getAnnotation(org.junit.Test.class);
348         return testAnnotation != null;
349     }

428     protected void printReportEntry(ReportEntry entry, PrintStream out) {
429         if (entry == null) {
430             out.println("    entry == null"); //$NON-NLS-1$
431             return;
432         }
433         out.println("    entry.getName():       " + quoteNonNullString(entry.getName())); //$NON-NLS-1$
434         out.println("         .getMessage():    " + quoteNonNullString(entry.getMessage())); //$NON-NLS-1$
435         if (entry.getStackTraceWriter() != null) {
436             out.println("         .getStackTraceWriter(): NOT null"); //$NON-NLS-1$
437         }
438     }

463     protected void sendMessage(String msg) {
464         if (fWriter == null) {
465             return;
466         }
467         if (fTraceSocketMessages) {
468             System.out.println(msg);
469         }
470         fWriter.println(msg);
471         fWriter.flush();
472     }
473 
474     private void sendStackTrace(ReportEntry entry) {
475         StackTraceWriter stackTraceWriter = entry.getStackTraceWriter();
476         if ((fWriter == null) || (stackTraceWriter == null)) {
477             return;
478         }
479         sendMessage(JUnitViewMessageIds.TRACE_START);
480         sendMessage(stackTraceWriter.writeTraceToString());
481         sendMessage(JUnitViewMessageIds.TRACE_END);
482     }

484     protected int getTestId(String testName) {
485         Integer id = fTestNameToIdMap.get(testName);
486         if (id != null) {
487             return id;
488         }
489         id = fNextAvailableId++;
490         fTestNameToIdMap.put(testName, id);
491         return id;
492     }

494     private static String escapeTestName(String s) {
495         if ((s.indexOf(',') < 0) && (s.indexOf('\\') < 0) && (s.indexOf('\r') < 0) && (s.indexOf('\n') < 0))
496             return s;
497         StringBuffer sb = new StringBuffer(s.length() + 10);
498         for (int i = 0; i < s.length(); i++) {
499             char c = s.charAt(i);
500             if (c == ',') {
501                 sb.append("\\,"); //$NON-NLS-1$
502             } else if (c == '\\') {
503                 sb.append("\\\\"); //$NON-NLS-1$
504             } else if (c == '\r') {
505                 if (i + 1 < s.length() && s.charAt(i + 1) == '\n') {
506                     i++;
507                 }
508                 sb.append(' ');
509             } else if (c == '\n') {
510                 sb.append(' ');
511             } else {
512                 sb.append(c);
513             }
514         }
515         return sb.toString();
516     }
