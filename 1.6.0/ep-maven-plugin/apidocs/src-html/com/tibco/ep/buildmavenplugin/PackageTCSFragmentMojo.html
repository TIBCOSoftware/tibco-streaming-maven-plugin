008package com.tibco.ep.buildmavenplugin;
009
010import org.apache.maven.artifact.Artifact;
011import org.apache.maven.plugin.MojoExecutionException;
012import org.apache.maven.plugins.assembly.model.Assembly;
013import org.apache.maven.plugins.assembly.model.FileItem;
014import org.apache.maven.plugins.assembly.model.FileSet;
015import org.apache.maven.plugins.annotations.Mojo;
016import org.apache.maven.plugins.annotations.Parameter;
017import org.apache.maven.plugins.annotations.ResolutionScope;
018
019import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
020
021import java.io.File;
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026
027/**
028 * <p>Build a TCS fragment</p>
029 * 
030 * <p>The packaging rules are as follows :-</p>
031 * <ol>
032 * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
033 *      <ul>
034 *              <li>Archiver-Version: Plexus Archiver</li>
035 *              <li>Built-By: build user</li>
036 *              <li>Build-Jdk: jdk version</li>
037 *              <li>Package-Title: project groupId and name</li>
038 *              <li>Package-Version: project version and buildNumber</li>
039 *              <li>Package-Vendor: project organization name (if set)</li>
040 *              <li>TIBCO-EP-Build-Product-Version: product version</li>
041 *      </ul>
042 * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
043 * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
044 * <li>The project's files are copied into /</li>
045 * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
046 * <li>The project's resources files are copied into /</li>
047 * </ol>
048 * 
049 * <p>The plexus archiver is used to create the archive via the maven assembly
050 * plugin.</p>
051 * 
052 * <p>Native archives (nar) are included - in this case a mapping is included
053 * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
054 * values.</p>
055 * 
056 * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-tcs-fragment.zip</p>
057 * 
058 */
059@Mojo(name = "package-tcs-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true)
060public class PackageTCSFragmentMojo extends BasePackageMojo {
061
062    /**
063     * <p>Additional resources directory for HOCON configurations</p>
064     * 
065     * <p>This is added to the list of resource directories</p>
066     * 
067     * <p>Example use in pom.xml:</p>
068     * <img src="uml/package-configurationDirectory.svg" alt="pom">
069     * 
070     * @since 1.0.0
071     */
072    @Parameter( defaultValue = "${project.basedir}/src/main/configurations", required = true )
073    File configurationDirectory;
074
075    /**
076     * <p>Additional resources directory for test HOCON configurations</p>
077     * 
078     * <p>This is added to the list of test resource directories</p>
079     * 
080     * <p>Example use in pom.xml:</p>
081     * <img src="uml/package-testConfigurationDirectory.svg" alt="pom">
082     * 
083     * @since 1.0.0
084     */
085    @Parameter( defaultValue = "${project.basedir}/src/test/configurations", required = true )
086    File testConfigurationDirectory;
087
088
089    public void execute() throws MojoExecutionException {
090        getLog().debug( "Creating TCS fragment" );
091
092        prechecks();
093
094        Assembly assembly = createAssembly();
095
096        // add runtime dependencies
097        //
098        // we add them one-by-one instead of using a dependencyset so we can
099        // set the classpath in the manifest file
100        //
101        String classPath = "";
102        String nativeClassPath = "";
103        String fragmentList = "";
104
105        List<FileItem> deps = new ArrayList<FileItem>();
106        for (Artifact dependency : getCompileProjectDependencies()) {
107            String type = dependency.getType();
108            String destName = dependency.getGroupId()+"-"+dependency.getArtifactId()+"-"+dependency.getBaseVersion();
109
110            if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
111                destName += "-"+type+".zip";
112            } else {
113                if (dependency.hasClassifier()) {
114                    destName += "-"+dependency.getClassifier();
115                }
116                destName += "."+dependency.getType();
117            }
118
119            FileItem dependencyFile = new FileItem();
120            dependencyFile.setOutputDirectory("");
121            dependencyFile.setSource(getArtifactPath(dependency));
122            dependencyFile.setDestName(destName);
123            deps.add(dependencyFile);
124
125            if (dependency.getType().equals("nar")) {
126                nativeClassPath += destName+" ";
127            } else if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
128                fragmentList += destName+" ";
129            } else {
130                classPath += destName+" ";
131            }
132        }
133
134        // build a manifest file and add as first file to the zip
135        //
136        Map<String, String> extras = new HashMap<String, String>();
137        extras.put("TIBCO-EP-Fragment-Format-Version", "2");
138        if (!classPath.isEmpty()) {
139            extras.put("Class-Path", classPath);
140        }
141        if (!nativeClassPath.isEmpty()) {
142            extras.put("TIBCO-EP-Native-Class-Path", nativeClassPath);
143        }
144        if (!fragmentList.isEmpty()) {
145            extras.put("TIBCO-EP-Fragment-List", fragmentList);
146        }            
147        File manifest = packageManifest(assembly, null, extras);
148
149        for (FileItem dependencyFile : deps) {
150            assembly.addFile(dependencyFile);
151        }
152
153        // add output directory
154        //
155        // this should include resources ( already copied here ) and
156        // eventflow files
157        //
158        FileSet fileSet = new FileSet();
159        fileSet.setDirectory(project.getBuild().getOutputDirectory());
160        fileSet.setOutputDirectory("");
161        assembly.addFileSet(fileSet);
162
163        // write assembly
164        //
165        writeAssembly(assembly);
166        if (!manifest.delete()) {
167            getLog().debug("Unable to delete "+manifest.getAbsolutePath());
168        }
169    }
170}
