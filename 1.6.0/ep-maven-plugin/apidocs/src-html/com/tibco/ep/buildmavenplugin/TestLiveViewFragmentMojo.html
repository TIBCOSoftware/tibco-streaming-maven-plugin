030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.model.PluginExecution;
033import org.apache.maven.model.Resource;
034import org.apache.maven.plugin.MojoExecutionException;
035import org.apache.maven.plugins.annotations.Mojo;
036import org.apache.maven.plugins.annotations.Parameter;
037
038import static org.apache.maven.plugins.annotations.LifecyclePhase.TEST;
039
040import java.io.File;
041import java.util.Properties;
042
043/**
044 * <p>Test a LiveView fragment using sbunit.</p>
045 * 
046 * <p>A java runner for surefire is created and deployed to the test nodes - 
047 * this runner invokes surefire and hence junit to run the test cases.</p>
048 * 
049 * <p>The java runner sets the system property <b>net.sourceforge.cobertura.datafile</b>
050 * to a unique file in the build directory.  So when coverage is enabled, these
051 * per-node reports are merged into a single report for processing.</p>
052 * 
053 * <p>The java runner takes the configured reportsDirectory property and
054 * appends the node name - the node specific test report files are created in
055 * this directory.  When all the test have finished, these node specific test
056 * reports are renamed into the configured reportsDirectory directory.</p>
057 * 
058 * <p>The maven property jenkins.executionId.reportsDirectory is set so that
059 * jenkins can locate the test reports.</p>
060 * 
061 * <p>Java assertions are always enabled.  Options file is always ignored.</p>
062 * 
063 * <p>The following system properties are set :</p>
064 * <ul>
065 * <li><b>com.tibco.ep.dtm.fragment.version</b> - fragment version</li>
066 * <li><b>com.tibco.ep.dtm.fragment.identifier</b> - fragment identifier</li>
067 * </ul>
068 */
069@Mojo(name = "test-liveview-fragment", defaultPhase = TEST, threadSafe = true)
070public class TestLiveViewFragmentMojo extends BaseTestMojo {
071    
072    /**
073     * <p>Eventflow source directories</p>
074     * 
075     * <p>If no eventflowDirectories is specified, a single directory of
076     * ${project.basedir}/src/main/eventflow is used.</p>
077     * 
078     * <p>Example use in pom.xml:</p>
079     * <img src="uml/eventflowDirectories.svg" alt="pom">
080     * 
081     * <p>Example use on commandline:</p>
082     * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
083     * 
084     * @since 1.0.0
085     */
086    @Parameter( required = false, property = "eventflowDirectories" )
087    File[] eventflowDirectories;
088    
089    /**
090     * <p>Liveview source directory</p>
091     * 
092     * <p>Example use in pom.xml:</p>
093     * <img src="uml/liveviewDirectory.svg" alt="pom">
094     * 
095     * <p>Example use on commandline:</p>
096     * <img src="uml/liveviewDirectory-commandline.svg" alt="pom">
097     * 
098     * @since 1.0.0
099     */
100    @Parameter( defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory" )
101    File liveviewDirectory;
102    
103    /**
104     * <p>Additional resources directory for HOCON configurations</p>
105     * 
106     * <p>This is added to the list of resource directories</p>
107     * 
108     * <p>Example use in pom.xml:</p>
109     * <img src="uml/package-configurationDirectory.svg" alt="pom">
110     * 
111     * @since 1.0.0
112     */
113    @Parameter( defaultValue = "${project.basedir}/src/main/configurations", required = true )
114    File configurationDirectory;
115
116    /**
117     * <p>Additional resources directory for test HOCON configurations</p>
118     * 
119     * <p>This is added to the list of test resource directories</p>
120     * 
121     * <p>Example use in pom.xml:</p>
122     * <img src="uml/package-testConfigurationDirectory.svg" alt="pom">
123     * 
124     * @since 1.0.0
125     */
126    @Parameter( defaultValue = "${project.basedir}/src/test/configurations", required = true )
127    File testConfigurationDirectory;
128    
129    public void execute() throws MojoExecutionException {
130        getLog().debug( "Testing live data mart fragment" );
131
132        if (eventflowDirectories == null || eventflowDirectories.length == 0) {
133            eventflowDirectories = new File[] { new File(project.getBasedir(), "/src/main/eventflow") };
134        }
135        
136        Properties modelProperties = project.getModel().getProperties();
137        boolean testCasesFound = true;
138        if (modelProperties.getProperty(TESTCASESFOUND_PROPERTY) != null && modelProperties.getProperty(TESTCASESFOUND_PROPERTY).equals("false")) {
139            testCasesFound = false;
140        }
141        
142        // determine if we have executions steps in the pom - if so we can skip id this
143        // run is a default one
144        //
145        boolean hasExecutions = false;
146        if (mojoExecution.getPlugin() != null && mojoExecution.getPlugin().getExecutions() != null) {
147            for (PluginExecution p : mojoExecution.getPlugin().getExecutions()) {
148                if (!p.getId().startsWith("default-")) {
149                    hasExecutions = true;
150                }
151            }
152        }
153        
154        if ((hasExecutions && mojoExecution.getExecutionId().startsWith("default-")) || skipTests || installOnly || !testCasesFound) {
155            getLog().info("Tests are skipped (skipTests="+skipTests+",hasExecutions="+hasExecutions+",installOnly="+installOnly+",testCasesFound="+testCasesFound+")");
156            return;
157        }
158        
159        prechecks();
160        
161        Resource resource = new Resource();
162        resource.setDirectory(configurationDirectory.getAbsolutePath());
163        resource.setTargetPath("configurations");
164        project.getBuild().addResource(resource);
165        resource = new Resource();
166        resource.setDirectory(testConfigurationDirectory.getAbsolutePath());
167        resource.setTargetPath("test-configuration");
168        project.getBuild().addTestResource(resource);  
169        
170        initializeAdministration(true);
171        
172        // run test cases
173        //
174        runJunitTests(true, eventflowDirectories, liveviewDirectory);
175    }
176
177}
