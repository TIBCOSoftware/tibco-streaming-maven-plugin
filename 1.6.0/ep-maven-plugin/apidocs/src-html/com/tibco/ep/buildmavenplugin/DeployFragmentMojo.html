030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.model.PluginExecution;
033import org.apache.maven.plugin.MojoExecutionException;
034import org.apache.maven.plugins.annotations.Mojo;
035import org.apache.maven.plugins.annotations.Parameter;
036
037import static org.apache.maven.plugins.annotations.LifecyclePhase.TEST;
038
039import java.io.File;
040import java.util.ArrayList;
041import java.util.Arrays;
042import java.util.List;
043import java.util.Map.Entry;
044
045/**
046 * <p>Deploy an application fragment</p>
047 * 
048 * <p>Example use in pom.xml:</p>
049 * <img src="uml/deploy-fragment1.svg" alt="pom">
050 * 
051 * <p>A liveview fragment requires fragmentType, target and project set:</p>
052 * 
053 * <img src="uml/deploy-fragment2.svg" alt="pom">
054 */
055@Mojo(name = "deploy-fragment", defaultPhase = TEST, threadSafe = true)
056public class DeployFragmentMojo extends BaseTestMojo {
057    
058    /**
059     * <p>Eventflow source directories</p>
060     * 
061     * <p>If no eventflowDirectories is specified, a single directory of
062     * ${project.basedir}/src/main/eventflow is used.</p>
063     * 
064     * <p>Example use in pom.xml:</p>
065     * <img src="uml/eventflowDirectories.svg" alt="pom">
066     * 
067     * @since 1.0.0
068     */
069    @Parameter( required = false, property = "eventflowDirectories" )
070    File[] eventflowDirectories;
071    
072    /**
073     * <p>Liveview source directory</p>
074     * 
075     * <p>Example use in pom.xml:</p>
076     * <img src="uml/liveviewDirectory.svg" alt="pom">
077     * 
078     * <p>Example use on commandline:</p>
079     * <img src="uml/liveviewDirectory-commandline.svg" alt="pom">
080     * 
081     * @since 1.0.0
082     */
083    @Parameter( defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory" )
084    File liveviewDirectory;
085    
086    /**
087     * <p>Fragment parameters</p>
088     * 
089     * <p>Example use in pom.xml:</p>
090     * <img src="uml/fragment-arguments.svg" alt="pom">
091     * 
092     * @since 1.0.0
093     */
094    @Parameter
095    String[] arguments;
096    
097    /**
098     * <p>Target to deploy</p>
099     * 
100     * <p>
101     * Target can a main class name, a JAR with a main class name in the manifest, a
102     * fully scoped EventFlow module name, an EventFlow sbapp or ssql file, or the
103     * keyword "liveview"</p>
104     * 
105     * <p>Example use in pom.xml:</p>
106     * <img src="uml/fragment-target.svg" alt="pom">
107     * 
108     * @since 1.0.0
109     */
110    @Parameter
111    String target;
112    
113    /**
114     * Fragment type
115     */
116    public enum FRAGMENT {
117        /**
118         * Java
119         */
120        JAVA,
121        /**
122         * StreamBase
123         */
124        STREAMBASE,
125        /**
126         * Liveview data mart
127         */
128        LIVEVIEW
129      }
130    
131    /**
132     * <p>Fragment type</p>
133     * 
134     * <p>Example use in pom.xml:</p>
135     * <img src="uml/fragment-fragmentType.svg" alt="pom">
136     * 
137     * <p>One of JAVA, STREAMBASE or LIVEVIEW</p>
138     */
139    @Parameter(defaultValue = "JAVA")
140    FRAGMENT fragmentType;
141
142    public void execute() throws MojoExecutionException {
143        getLog().debug("Deploy");
144
145        if (eventflowDirectories == null) {
146            eventflowDirectories = new File[] { new File(project.getBasedir(), "/src/main/eventflow") };
147        }
148        
149        // determine if we have executions steps in the pom - if so we can skip id this
150        // run is a default one
151        //
152        boolean hasExecutions = false;
153        if (mojoExecution.getPlugin() != null && mojoExecution.getPlugin().getExecutions() != null) {
154            for (PluginExecution p : mojoExecution.getPlugin().getExecutions()) {
155                if (!p.getId().startsWith("default-")) {
156                    hasExecutions = true;
157                }
158            }
159        }
160
161        if ((hasExecutions && mojoExecution.getExecutionId().startsWith("default-")) || skipTests) {
162            getLog().info("Deploy is skipped (skipTests="+skipTests+",hasExecutions="+hasExecutions+")");
163            return;
164        }
165        
166        prechecks();
167        
168        initializeAdministration(true);
169
170        List<String> applArguments = new ArrayList<>();
171
172        if ((arguments != null) && (arguments.length != 0)) {
173            applArguments.addAll(Arrays.asList(arguments));
174        }
175
176        String thisServiceName=serviceName;
177        if (serviceName == null ||serviceName.length() == 0) {
178            thisServiceName = clusterName;
179        }
180        
181        // form list of java options - set in pom + command line
182        //
183        ArrayList<String> fullJavaOptions = new ArrayList<String>();
184        if (javaOptions != null && javaOptions.length > 0) {
185            fullJavaOptions.addAll(Arrays.asList(javaOptions));
186        }
187        if (optionsProperty != null && optionsProperty.length > 0) {
188            fullJavaOptions.addAll(Arrays.asList(optionsProperty));
189        }
190        if (systemPropertyVariables != null && systemPropertyVariables.size() >0) {
191            for (Entry<String, String> entry : systemPropertyVariables.entrySet()) {
192                fullJavaOptions.add("-D"+entry.getKey() + "=" + entry.getValue());
193            }
194        }
195        deploy(true, fragmentType.toString(), target, thisServiceName, userName, password, fullJavaOptions.toArray(new String[fullJavaOptions.size()]), applArguments, eventflowDirectories, liveviewDirectory, false);
196
197    }
198
