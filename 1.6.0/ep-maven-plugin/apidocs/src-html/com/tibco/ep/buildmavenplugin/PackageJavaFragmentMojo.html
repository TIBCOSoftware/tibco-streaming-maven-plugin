030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.artifact.Artifact;
033import org.apache.maven.plugin.MojoExecutionException;
034import org.apache.maven.plugins.annotations.Mojo;
035import org.apache.maven.plugins.annotations.Parameter;
036
037import org.apache.maven.plugins.assembly.model.Assembly;
038import org.apache.maven.plugins.assembly.model.FileItem;
039import org.apache.maven.plugins.assembly.model.FileSet;
040
041import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
042
043import java.io.File;
044import java.util.ArrayList;
045import java.util.HashMap;
046import java.util.List;
047import java.util.Map;
048
049/**
050 * <p>Build a java fragment zip archive.</p>
051 *
052 * <p>The packaging rules are as follows :-</p>
053 * <ol>
054 * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
055 *      <ul>
056 *              <li>Archiver-Version: Plexus Archiver</li>
057 *              <li>Built-By: build user</li>
058 *              <li>Build-Jdk: jdk version</li>
059 *              <li>Package-Title: project groupId and name</li>
060 *              <li>Package-Version: project version and buildNumber</li>
061 *              <li>Package-Vendor: project organization name (if set)</li>
062 *              <li>Main-Class: mainClass parameter (if set)</li>
063 *              <li>TIBCO-EP-Build-Product-Version: product version</li>
064 *      </ul>
065 * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
066 * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
067 * <li>The project's built java classes files are copied into /classes</li>
068 * <li>The project's resource files are copied into /</li>
069 * <li>Java runtime dependencies are copied into /dependencies/java/</li>
070 * </ol>
071 * 
072 * <p>The plexus archiver is used to create the archive via the maven assembly
073 * plugin.</p>
074 * 
075 * <p>Native archives (nar) are included - in this case a mapping is included
076 * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
077 * values.</p>
078 * 
079 * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-java-fragment.zip</p>
080 */
081@Mojo(name = "package-java-fragment", defaultPhase = PACKAGE, threadSafe = true)
082public class PackageJavaFragmentMojo extends BasePackageMojo {
083
084    // maven user parameters
085    //
086
087    /**
088     * <p>Java Main class name.  This will be added to the MANIFEST file.</p>
089     * 
090     * <p>Example use in pom.xml:</p>
091     * <img src="uml/package-mainClass.svg" alt="pom">
092     * 
093     * @since 1.0.0
094     */
095    @Parameter
096    String mainClass;
097
098    public void execute() throws MojoExecutionException {
099        getLog().debug( "Creating fragment zip" );
100
101        prechecks();
102
103        // updated format which is really jar
104        //
105
106        Assembly assembly = createAssembly();
107
108        // add runtime dependencies
109        //
110        // we add them one-by-one instead of using a dependencyset so we can
111        // set the classpath in the manifest file
112        //
113        String classPath = "";
114        String nativeClassPath = "";
115        String fragmentList = "";
116
117        List<FileItem> deps = new ArrayList<FileItem>();
118        for (Artifact dependency : getCompileProjectDependencies()) {
119            String type = dependency.getType();
120            String destName = dependency.getGroupId()+"-"+dependency.getArtifactId()+"-"+dependency.getBaseVersion();
121
122            if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
123                destName += "-"+type+".zip";
124            } else {
125                if (dependency.hasClassifier()) {
126                    destName += "-"+dependency.getClassifier();
127                }
128                destName += "."+dependency.getType();
129            }
130
131            FileItem dependencyFile = new FileItem();
132            dependencyFile.setOutputDirectory("");
133            dependencyFile.setSource(getArtifactPath(dependency));
134            dependencyFile.setDestName(destName);
135            deps.add(dependencyFile);
136
137            if (dependency.getType().equals("nar")) {
138                nativeClassPath += destName+" ";
139            } else if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
140                fragmentList += destName+" ";
141            } else {
142                classPath += destName+" ";
143            }
144        }
145
146        // build a manifest file and add as first file to the zip
147        //
148        Map<String, String> extras = new HashMap<String, String>();
149        extras.put("TIBCO-EP-Fragment-Format-Version", "2");
150        if (!classPath.isEmpty()) {
151            extras.put("Class-Path", classPath);
152        }
153        if (!nativeClassPath.isEmpty()) {
154            extras.put("TIBCO-EP-Native-Class-Path", nativeClassPath);
155        }
156        if (!fragmentList.isEmpty()) {
157            extras.put("TIBCO-EP-Fragment-List", fragmentList);
158        }
159        File manifest = packageManifest(assembly, mainClass, extras);
160
161        for (FileItem dependencyFile : deps) {
162            assembly.addFile(dependencyFile);
163        }
164
165        // add output directory
166        //
167        // this should include resources ( already copied here ) and
168        // eventflow files
169        //
170        FileSet fileSet = new FileSet();
171        fileSet.setDirectory(project.getBuild().getOutputDirectory());
172        fileSet.setOutputDirectory("");
173        assembly.addFileSet(fileSet);
174
175        // write assembly
176        //
177        writeAssembly(assembly);
178        if (!manifest.delete()) {
179            getLog().debug("Unable to delete "+manifest.getAbsolutePath());
180        }
181    }
182
183}
