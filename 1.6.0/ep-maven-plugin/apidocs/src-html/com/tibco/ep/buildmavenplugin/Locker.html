001package com.tibco.ep.buildmavenplugin;
002
003import java.io.File;
004import java.io.IOException;
005import java.io.RandomAccessFile;
006import java.nio.channels.FileChannel;
007import java.nio.channels.FileLock;
008import java.nio.channels.FileLockInterruptionException;
009import java.nio.channels.OverlappingFileLockException;
010
011/**
012 * Utility class to lock a file / directory
013 *
014 */
015public class Locker {
016    private RandomAccessFile m_raf;
017    private FileChannel m_fc;
018    private FileLock m_fl;
019    
020    /**
021     * create a lock - if already locked block
022     * @param f file or directory to lock
023     */
024    public Locker(File f) {
025
026        // suport locking directories
027        //
028        File lockFile = f;
029        if (f.isDirectory()) {
030            lockFile = new File(f, ".lock");
031            if (!lockFile.exists()) {
032                try {
033                    lockFile.createNewFile();
034                } catch (IOException e1) {
035                }
036            }
037        }
038        
039        boolean locked = false;
040        while (!locked) {
041            try {
042                m_raf = new RandomAccessFile(lockFile, "rw");
043                m_fc = m_raf.getChannel();
044                m_fl = m_fc.lock();
045                locked=true;
046            } catch (FileLockInterruptionException e) {
047                try {
048                    Thread.sleep(200);
049                } catch (InterruptedException e1) {
050                }
051            } catch (OverlappingFileLockException e) {
052                try {
053                    Thread.sleep(200);
054                } catch (InterruptedException e1) {
055                }
056            } catch (IOException e) {
057                System.out.println(e.toString());
058                m_raf = null;
059                m_fc = null;
060                m_fl = null;
061            }
062        }
063    }
064
065    /**
066     * release lock
067     */
068    public void release() {
069        if (m_fl != null) {
070            try {
071                m_fl.release();
072            }
073            catch (IOException e) {
074                System.out.println(e.toString());
075            }
076        }
077    }
078}
