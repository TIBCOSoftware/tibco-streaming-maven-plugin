030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.model.Resource;
033import org.apache.maven.plugin.MojoExecutionException;
034import org.apache.maven.plugins.annotations.Mojo;
035import org.apache.maven.plugins.annotations.Parameter;
036
037import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
038
039import java.io.File;
040
041/**
042 * <p>Set resources</p>
043 *
044 */
045@Mojo(name = "set-resources", defaultPhase = PACKAGE, threadSafe = true)
046public class SetResources extends BasePackageMojo {
047
048    // maven user parameters
049    //
050
051    /**
052     * <p>Additional resources directory for HOCON configurations</p>
053     * 
054     * <p>This is added to the list of resource directories</p>
055     * 
056     * <p>Example use in pom.xml:</p>
057     * <img src="uml/package-configurationDirectory.svg" alt="pom">
058     * 
059     * @since 1.0.0
060     */
061    @Parameter( defaultValue = "${project.basedir}/src/main/configurations", required = true )
062    File configurationDirectory;
063
064    /**
065     * <p>Additional resources directory for test HOCON configurations</p>
066     * 
067     * <p>This is added to the list of test resource directories</p>
068     * 
069     * <p>Example use in pom.xml:</p>
070     * <img src="uml/package-testConfigurationDirectory.svg" alt="pom">
071     * 
072     * <p>Example use on commandline:</p>
073     * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
074     * 
075     * @since 1.0.0
076     */
077    @Parameter( defaultValue = "${project.basedir}/src/test/configurations", required = true )
078    File testConfigurationDirectory;
079
080    /**
081     * <p>Eventflow directories</p>
082     * 
083     * <p>This is added to the list of resource directories</p>
084     * 
085     * <p>Example use on commandline:</p>
086     * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
087     * 
088     * @since 1.1.0
089     */
090    @Parameter( required = false, property = "eventflowDirectories" )
091    File[] eventflowDirectories;
092
093    /**
094     * <p>Liveview source directory</p>
095     * 
096     * <p>This is added to the list of resource directories</p>
097     * 
098     * <p>Example use in pom.xml:</p>
099     * <img src="uml/liveviewDirectory.svg" alt="pom">
100     * 
101     * <p>Example use on commandline:</p>
102     * <img src="uml/liveviewDirectory-commandline.svg" alt="pom">
103     * 
104     * @since 1.1.0
105     */
106    @Parameter( defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory" )
107    File liveviewDirectory;
108
109    public void execute() throws MojoExecutionException {
110        getLog().debug( "Set resources" );
111
112        prechecks();
113
114        if (eventflowDirectories == null || eventflowDirectories.length == 0) {
115            eventflowDirectories = new File[] { new File(project.getBasedir(), "/src/main/eventflow") };
116        }
117        
118        if (configurationDirectory.exists()) {
119            Resource resource = new Resource();
120            resource.setDirectory(configurationDirectory.getAbsolutePath());
121            //resource.setTargetPath(project.getBasedir().toPath().relativize(configurationDirectory.toPath()).toString());
122            project.getBuild().addResource(resource);
123        }
124        
125        if (testConfigurationDirectory.exists()) {
126            Resource resource = new Resource();
127            resource.setDirectory(testConfigurationDirectory.getAbsolutePath());
128            //resource.setTargetPath(project.getBasedir().toPath().relativize(testConfigurationDirectory.toPath()).toString());
129            project.getBuild().addTestResource(resource);
130        }
131        
132        for (File ef : eventflowDirectories) {
133            if (ef.exists()) {
134                Resource resource = new Resource();
135                resource.setDirectory(ef.getAbsolutePath());
136                //resource.setTargetPath(project.getBasedir().toPath().relativize(ef.toPath()).toString());
137                project.getBuild().addResource(resource);
138            }
139        }
140
141        if (liveviewDirectory.exists()) {
142            Resource resource = new Resource();
143            resource.setDirectory(liveviewDirectory.getAbsolutePath());
144            //resource.setTargetPath(project.getBasedir().toPath().relativize(liveviewDirectory.toPath()).toString());
145            project.getBuild().addResource(resource);
146        }
147        
148        if (this.ignoreLeaks != null && this.ignoreLeaks.length >0) {
149            String ignoreLeaks = String.join(",", this.ignoreLeaks);
150            project.getProperties().setProperty("ignoreLeaks", ignoreLeaks);
151            getLog().debug("ignoreLeaks="+ignoreLeaks);
152        }
153        
154        
155    }
156
157}
