030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.logging.Log;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036/**
037 * Simulated maven logger to catch the actual output
038 * 
039 */
040public class SimulatedLog implements Log {
041    private Logger logger = LoggerFactory.getLogger(SimulatedLog.class);
042
043    private final boolean verbose;
044    private String debugText;
045    private String infoText;
046    private String errorText;
047    private String warnText;
048
049    /**
050     * Constructor
051     * @param verbose true to log all output
052     */
053    SimulatedLog(boolean verbose) {
054        this.verbose = verbose;
055        reset();
056    }
057    
058    /**
059     * Reset stored stings
060     */
061    public void reset() {
062        debugText = "";
063        infoText = "";
064        errorText = "";
065        warnText = "";
066    }
067    
068    /**
069     * Get stored debug log
070     * 
071     * @return debug log
072     */
073    public String getDebugLog() {
074        return debugText;
075    }
076
077    /**
078     * Get stored info log
079     * 
080     * @return info log
081     */
082    public String getInfoLog() {
083        return infoText;
084    }
085    
086    /**
087     * Get stored error log
088     * 
089     * @return error log
090     */
091    public String getErrorLog() {
092        return errorText;
093    }
094
095    /**
096     * Get stored warn log
097     * 
098     * @return warn log
099     */
100    public String getWarnLog() {
101        return warnText;
102    }
103    
104    @Override
105    public void debug(CharSequence content) {
106        if (verbose) {
107            logger.info("DEBUG: "+content);
108        }
109        this.debugText+=content+"\n";
110    }
111
112    @Override
113    public void debug(Throwable error) {
114        if (verbose) {
115            logger.info("DEBUG: "+error);
116        }
117        this.debugText+=error+"\n";
118    }
119
120    @Override
121    public void debug(CharSequence content, Throwable error) {
122        if (verbose) {
123            logger.info("DEBUG: "+content+error);
124        }
125        this.debugText+=content+"\n";
126        this.debugText+=error+"\n";
127    }
128
129    @Override
130    public void error(CharSequence content) { 
131        if (verbose) {
132            logger.info("ERROR: "+content);
133        }
134        this.errorText+=content+"\n";
135    }
136
137    @Override
138    public void error(Throwable error) {
139        if (verbose) {
140            logger.info("ERROR: "+error);
141        }
142        this.errorText+=error+"\n";
143    }
144
145    @Override
146    public void error(CharSequence content, Throwable error) { 
147        if (verbose) {
148            logger.info("ERROR: "+content+error);
149        }
150        this.errorText+=content+"\n";
151        this.errorText+=error+"\n";
152    }
153
154    @Override
155    public void info(CharSequence content) {
156        if (verbose) {
157            logger.info("INFO: "+content);
158        }
159        this.infoText+=content+"\n";
160
161    }
162
163    @Override
164    public void info(Throwable error) {
165        if (verbose) {
166            logger.info("INFO: "+error);
167        }
168        this.infoText+=error+"\n";
169    }
170
171    @Override
172    public void info(CharSequence content, Throwable error) {
173        if (verbose) {
174            logger.info("INFO: "+content+error);
175        }
176        this.infoText+=content+"\n";
177        this.infoText+=error+"\n";
178    }
179
180    @Override
181    public boolean isDebugEnabled() {
182        return true;
183    }
184
185    @Override
186    public boolean isErrorEnabled() {
187        return true;
188    }
189
190    @Override
191    public boolean isInfoEnabled() {
192        return true;
193    }
194
195    @Override
196    public boolean isWarnEnabled() {
197        return true;
198    }
199
200    @Override
201    public void warn(CharSequence content) {
202        if (verbose) {
203            logger.info("WARN: "+content);
204        }
205        this.warnText+=content+"\n";
206
207    }
208
209    @Override
210    public void warn(Throwable error) {
211        if (verbose) {
212            logger.info("WARN: "+error);
213        }
214        this.warnText+=error+"\n";
215
216    }
217
218    @Override
219    public void warn(CharSequence content, Throwable error) {
220        if (verbose) {
221            logger.info("WARN: "+content+error);
222        }
223        this.warnText+=content+"\n";
224        this.warnText+=error+"\n";
225    }
226
227}
