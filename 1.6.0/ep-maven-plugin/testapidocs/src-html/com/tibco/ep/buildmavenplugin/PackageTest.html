380    /**
381     * Check if the zip  contains a file
382     * @param zip zip archive
383     * @param filename filename
384     * @return true if it does contain it
385     */
386    boolean zipContains(String zip, String filename) {
387        boolean found=false;
388
389        ZipFile zipFile;
390        try {
391            zipFile = new ZipFile(zip);
392
393            Enumeration<? extends ZipEntry> zipEntries = zipFile.entries();
394            String fname;
395            while (zipEntries.hasMoreElements()) {
396                fname = ((ZipEntry)zipEntries.nextElement()).getName();
397                if (fname.equals(filename)) {
398                    found = true;
399                }
400            }
401            zipFile.close();
402        } catch (IOException e) {
403            // TODO Auto-generated catch block
404            e.printStackTrace();
405        }
406
407        return found;
408    }

410    /**
411     * Check if the file contains test
412     * @param filename gile
413     * @param text text to look for
414     * @return true if it does contain it
415     */
416    boolean fileContains(String filename, String text) {
417        boolean found = false;
418        try {
419            File file = new File(filename);
420            Scanner scanner = new Scanner(file);
421            while (scanner.hasNextLine()) {
422                String line = scanner.nextLine();
423                if(line.contains(text)) {
424                     found = true;
425                }
426            }
427            scanner.close();
428        } catch (FileNotFoundException e) {
429        }
430        return found;
431    }
