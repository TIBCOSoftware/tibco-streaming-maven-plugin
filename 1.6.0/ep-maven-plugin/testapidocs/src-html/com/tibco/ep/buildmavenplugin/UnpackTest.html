030package com.tibco.ep.buildmavenplugin;
031
032import java.io.File;
033import org.apache.maven.plugin.testing.MojoRule;
034import org.apache.maven.plugin.testing.resources.TestResources;
035import org.junit.Assert;
036import org.junit.Rule;
037import org.junit.Test;
038
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042import com.tibco.ep.buildmavenplugin.InstallProductMojo;
043import com.tibco.ep.buildmavenplugin.UnpackFragmentMojo;
044import com.tibco.ep.buildmavenplugin.UnpackNarMojo;
045
046/**
047 * Unpack tests
048 *
049 */
050public class UnpackTest extends BetterAbstractMojoTestCase  {
051    private Logger logger = LoggerFactory.getLogger(UnpackTest.class);
052
053    /**
054     * rule
055     */
056    @Rule
057    public MojoRule rule = new MojoRule();
058
059    /**
060     * resources
061     */
062    @Rule
063    public TestResources resources = new TestResources();
064
065    /**
066     * Install product - do this first
067     * 
068     * @throws Exception on error
069     */
070    @Test
071    public void testUnpack() throws Exception  {    
072        logger.info("Install Product");
073        SimulatedLog simulatedLog = new SimulatedLog(false);
074
075        File pom = new File( "target/projects", "pom.xml" );
076        Assert.assertNotNull( pom );
077        Assert.assertTrue( pom.exists());
078
079        InstallProductMojo installProduct = (InstallProductMojo) lookupConfiguredMojo(pom, "install-product");
080        Assert.assertNotNull( installProduct );
081        installProduct.setLog(simulatedLog);     
082        installProduct.execute();
083        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
084        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
085
086        logger.info("unpack test");
087        simulatedLog = new SimulatedLog(false);
088
089        File testPom = new File( "target/projects/eventflow", "pom.xml" );
090        Assert.assertNotNull(testPom);
091        Assert.assertTrue(testPom.exists());
092
093        // FIX THIS - we need to figure out how to add a dependency here so it gets extracted
094        
095        logger.info("   Unpack fragment");
096        UnpackFragmentMojo unpackFragment = (UnpackFragmentMojo) lookupConfiguredMojo(testPom, "unpack-fragment");
097        Assert.assertNotNull(unpackFragment);  
098        simulatedLog.reset();
099        unpackFragment.setLog(simulatedLog);
100        unpackFragment.execute();
101        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
102        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
103
104        logger.info("   Unpack nar");
105        UnpackNarMojo unpackNar = (UnpackNarMojo) lookupConfiguredMojo(testPom, "unpack-nar");
106        Assert.assertNotNull(unpackNar);  
107        simulatedLog.reset();
108        unpackNar.setLog(simulatedLog);
109        unpackNar.execute();
110        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
111        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
112
113    }
114
115}
