31  package com.tibco.ep.buildmavenplugin.admin;
32  
33  import com.tibco.ep.buildmavenplugin.ErrorHandling;
34  import com.tibco.ep.sb.services.management.INotifier;
35  import org.apache.maven.plugin.logging.Log;
36  
37  import java.util.List;
38  import java.util.Optional;
39  
40  /**
41   * The command notifier implementation
42   */
43  public class RuntimeCommandNotifier implements INotifier {
44  
45      private final Log log;
46      private final String command;
47      private final String header;
48      private boolean running;
49      private Optional<StringBuilder> results;
50      private ErrorHandling errorHandling;
51  
52      /**
53       * @param log The Maven logger
54       * @param command The command description string
55       * @param targetLocation The target location for reporting
56       */
57      public RuntimeCommandNotifier(Log log, String command, String targetLocation) {
58          this.log = log;
59          this.command = command;
60          this.header = "[" + targetLocation + "] ";
61  
62          //  FIX THIS (FL): for some reason, the DtmDeploy command does not call start.
63          //
64          this.running = true;
65  
66          errorHandling(ErrorHandling.FAIL);
67          recordOutput(false);
68      }
69  
70      /**
71       * @param errorHandling Error handling (to drive logging severity)
72       * @return This
73       */
74      public RuntimeCommandNotifier errorHandling(ErrorHandling errorHandling) {
75          this.errorHandling = errorHandling;
76          return this;
77      }
78  
79      /**
80       * @param recordOutput True if the full output should be available in {@link #getOutput()}
81       * @return This
82       */
83      public RuntimeCommandNotifier recordOutput(Boolean recordOutput) {
84          this.results = recordOutput ? Optional.of(new StringBuilder()) : Optional.empty();
85          return this;
86      }
87  
88      /**
89       * @return The command output
90       */
91      public String getOutput() {
92          assert results.isPresent();
93          return results.get().toString();
94      }
95  
96      @Override
97      public void start() {
98          logInfo("Running \"" + command + "\"");
99          running = true;
100     }
101 
102     @Override
103     public void cancel() {
104         logError("cancelled");
105     }
106 
107     @Override
108     public void failed(int returnCode) {
109         if (!running) {
110             return;
111         }
112 
113         logErrorIfNeeded("return code " + returnCode);
114         results.ifPresent(sb -> sb.append(returnCode).append("\n"));
115     }
116 
117     @Override
118     public void complete() {
119         if (!running) {
120             return;
121         }
122         logInfo("Finished \"" + command + "\"");
123         running = false;
124     }
125 
126     @Override
127     public void results(List<String> resultLines) {
128         if (!running) {
129             log.info("Received logs while not running: " + resultLines.size());
130             return;
131         }
132 
133         resultLines.forEach(this::logInfo);
134     }
135 
136     @Override
137     public void info(String source, String message) {
138         if (!running) {
139             return;
140         }
141         log.info("[" + source + "] " + message);
142         results.ifPresent(sb -> sb.append(message).append("\n"));
143     }
144 
145     @Override
146     public void error(String source, String message) {
147         if (!running) {
148             return;
149         }
150         log.error("[" + source + "] " + message);
151         results.ifPresent(sb -> sb.append(message).append("\n"));
152     }
153 
154     private void logErrorIfNeeded(String message) {
155         if (errorHandling == ErrorHandling.FAIL) {
156             logError(message);
157         } else {
158             logInfo(message + " [ignored]");
159         }
160     }
161 
162     private void logInfo(String message) {
163         log.info(header + message);
164     }
165 
166     private void logError(String message) {
167         log.error(header + message);
168     }
169 }
