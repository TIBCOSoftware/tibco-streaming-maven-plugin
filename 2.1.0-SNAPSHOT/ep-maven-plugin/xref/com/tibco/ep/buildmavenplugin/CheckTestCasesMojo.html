30  package com.tibco.ep.buildmavenplugin;
31  
32  import org.apache.maven.plugin.MojoExecutionException;
33  import org.apache.maven.plugins.annotations.Mojo;
34  import org.apache.maven.plugins.annotations.Parameter;
35  
36  import static org.apache.maven.plugins.annotations.LifecyclePhase.TEST;
37  
38  import java.util.Properties;
39  
40  /**
41   * <p>Check for test cases.</p>
42   * 
43   * <p>If no test cases are found, <b>testCasesFound</b> property is set to true so that nodes are
44   * not started or stopped.</p>
45   */
46  @Mojo(name = "check-testcases", defaultPhase = TEST, threadSafe = true)
47  public class CheckTestCasesMojo extends BaseTestMojo {
48  
49      /**
50       * <p>Set this to 'true' to skip running tests, but still compile them.</p>
51       * 
52       * <p>Example use in pom.xml:</p>
53       * <img src="uml/start-nodes-skipTests.svg" alt="pom">
54       * 
55       * <p>Example use on commandline:</p>
56       * <img src="uml/start-nodes-skipTests-commandline.svg" alt="pom">
57       * 
58       * @since 1.0.0
59       */
60      @Parameter(property = "skipTests", defaultValue = "false")
61      boolean skipTests;
62      
63      public void execute() throws MojoExecutionException {
64  
65          getLog().debug( "Checking for test cases");
66  
67          if (skipTests || installOnly) {
68              getLog().info("Tests are skipped (skipTests="+skipTests+",installOnly="+installOnly+")");
69              return;
70          }
71          
72          Properties modelProperties = project.getModel().getProperties();
73          
74          String[] testClasses = scanForTests();
75          if (testClasses == null || testClasses.length == 0) {
76              getLog().info("No test cases found, setting testCasesFound=false to skip redundant start and stop nodes");
77              modelProperties.setProperty(TESTCASESFOUND_PROPERTY, "false");
78          } else {
79              modelProperties.setProperty(TESTCASESFOUND_PROPERTY, "true");
80          }
81          
82          project.getModel().setProperties(modelProperties);
83  
84      }
85  
86  }
