078    @Test
079    public void testUnitTest() throws Exception {
080        LOGGER.info("Install Product");
081        SimulatedLog simulatedLog = new SimulatedLog(false);
082
083        File pom = new File("target/projects", "pom.xml");
084        Assert.assertNotNull(pom);
085        Assert.assertTrue(pom.exists());
086
087        InstallProductMojo installProduct = (InstallProductMojo) lookupConfiguredMojo(pom, "install-product");
088        Assert.assertNotNull(installProduct);
089        installProduct.setLog(simulatedLog);
090        installProduct.execute();
091        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
092        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
093
094        // install to a different directory
095        //
096        File productHome = new File(Paths.get("").toAbsolutePath().toFile(), "producthome");
097        File markerFile;
098        if (System.getProperty("os.name").toLowerCase().indexOf("mac") >= 0) {
099            markerFile = new File(new File(productHome, "dependency-maven-plugin-markers"), "com.tibco.ep.dtm-platform_osxx86_64-zip-2.1.0-SNAPSHOT.marker");
100        } else if (System.getProperty("os.name").toLowerCase().indexOf("linux") >= 0) {
101            if (System.getProperty("os.arch").equals("aarch64")) {
102                markerFile = new File(new File(productHome, "dependency-maven-plugin-markers"), "com.tibco.ep.dtm-platform_linuxaarch64-zip-2.1.0-SNAPSHOT.marker");
103            } else {
104                markerFile = new File(new File(productHome, "dependency-maven-plugin-markers"), "com.tibco.ep.dtm-platform_linuxx86_64-zip-2.1.0-SNAPSHOT.marker");
105            }
106        } else {
107            markerFile = new File(new File(productHome, "dependency-maven-plugin-markers"), "com.tibco.ep.dtm-platform_windowsx64-zip-2.1.0-SNAPSHOT.marker");
108        }
109        installProduct = (InstallProductMojo) lookupConfiguredMojo(pom, "install-product");
110        Assert.assertNotNull(installProduct);
111        installProduct.productHome = productHome;
112        try {
113            Files.walkFileTree(productHome.toPath(), new SimpleFileVisitor<Path>() {
114                @Override
115                public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
116                    Files.delete(file);
117                    return FileVisitResult.CONTINUE;
118                }
119
120                @Override
121                public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
122                    Files.delete(dir);
123                    return FileVisitResult.CONTINUE;
124                }
125
126            });
127        } catch (IOException e) {
128            // ignore
129        }
130        installProduct.setLog(simulatedLog);
131        installProduct.execute();
132        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
133        assertTrue(markerFile.toString(), markerFile.exists());
134
135        // save good md5
136        //
137        String goodmd5 = "good not read";
138        BufferedReader input = null;
139        try {
140            input = new BufferedReader(new FileReader(markerFile));
141            goodmd5 = input.readLine();
142        } catch (IOException e) {
143        } finally {
144            if (input != null) {
145                input.close();
146            }
147        }
148
149        // broken md5
150        //
151        BufferedWriter output = null;
152        try {
153            output = new BufferedWriter(new FileWriter(markerFile));
154            output.write("junk");
155        } catch (IOException e) {
156        } finally {
157            if (output != null) {
158                try {
159                    output.close();
160                } catch (IOException e) {
161                }
162            }
163        }
164        installProduct = (InstallProductMojo) lookupConfiguredMojo(pom, "install-product");
165        Assert.assertNotNull(installProduct);
166        installProduct.productHome = productHome;
167        installProduct.setLog(simulatedLog);
168        installProduct.execute();
169        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
170        assertTrue(markerFile.exists());
171        String reinstallmd5 = "updated not read";
172        input = null;
173        try {
174            input = new BufferedReader(new FileReader(markerFile));
175            reinstallmd5 = input.readLine();
176        } catch (IOException e) {
177        } finally {
178            if (input != null) {
179                input.close();
180            }
181        }
182        assertEquals(reinstallmd5, goodmd5);
183
184        //  Java test.
185        //
186        LOGGER.info("java unit test");
187        simulatedLog = new SimulatedLog(true);
188
189        File testPom = new File("target/projects/java", "test.xml");
190        Assert.assertNotNull(testPom);
191        Assert.assertTrue(testPom.exists());
192
193        File startPom = new File("target/projects/java", "start.xml");
194        Assert.assertNotNull(startPom);
195        Assert.assertTrue(startPom.exists());
196
197        File stopPom = new File("target/projects/java", "stop.xml");
198        Assert.assertNotNull(stopPom);
199        Assert.assertTrue(stopPom.exists());
200
201        File deployPom = new File("target/projects/java", "deploy.xml");
202        Assert.assertNotNull(deployPom);
203        Assert.assertTrue(deployPom.exists());
204
205        runJavaTest(simulatedLog, testPom, startPom, stopPom, deployPom);
206
207        //  Event Flow test.
208        //
209        LOGGER.info("EventFlow unit test");
210        simulatedLog = new SimulatedLog(true);
211
212        testPom = new File("target/projects/eventflow", "pom.xml");
213        Assert.assertNotNull(testPom);
214        Assert.assertTrue(testPom.exists());
215        runEventFlowTest(simulatedLog, testPom);
216
217        //  LiveView test.
218        //
219        LOGGER.info("LiveView unit test");
220        simulatedLog = new SimulatedLog(false);
221
222        testPom = new File("target/projects/liveview", "pom.xml");
223        Assert.assertNotNull(testPom);
224        Assert.assertTrue(testPom.exists());
225
226        runLiveViewTest(simulatedLog, testPom);
227
228        LOGGER.info("application test");
229        simulatedLog = new SimulatedLog(false);
230
231        testPom = new File("target/projects/application", "pom.xml");
232        Assert.assertNotNull(testPom);
233        Assert.assertTrue(testPom.exists());
234
235        deployPom = new File("target/projects/application", "deploy.xml");
236        Assert.assertNotNull(testPom);
237        Assert.assertTrue(testPom.exists());
238
239        runApplicationTest(simulatedLog, testPom, deployPom);
240    }

242    private void runApplicationTest(SimulatedLog simulatedLog, File testPom, File deployPom) throws Exception {
243        LOGGER.info("   Start nodes");
244        StartNodesMojo startNodes = (StartNodesMojo) lookupConfiguredMojo(deployPom, "start-nodes");
245        Assert.assertNotNull(startNodes);
246        startNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
247        simulatedLog.reset();
248        startNodes.setLog(simulatedLog);
249        startNodes.execute();
250        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
251        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
252        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Node started"));
253
254
255        LOGGER.info("   Stop nodes");
256        StopNodesMojo stopNodes = (StopNodesMojo) lookupConfiguredMojo(testPom, "stop-nodes");
257        Assert.assertNotNull(stopNodes);
258        stopNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
259        simulatedLog.reset();
260        stopNodes.setLog(simulatedLog);
261        stopNodes.execute();
262        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
263        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
264        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Node removed"));
265    }

343    private void runJavaTest(SimulatedLog simulatedLog, File testPom, File startPom, File stopPom, File deployPom) throws Exception {
344
345        // copy test file
346        //
347        try {
348            new File("target/projects/java/target/test-classes/com/tibco/ep/buildmavenplugin")
349                .mkdirs();
350            Files
351                .copy(new File("target/test-classes/com/tibco/ep/buildmavenplugin/DummyTest.class")
352                        .toPath(),
353                    new File("target/projects/java/target/test-classes/com/tibco/ep/buildmavenplugin/DummyTest.class")
354                        .toPath());
355        } catch (FileAlreadyExistsException e) {
356            // ignore
357        }
358
359        new File("target/projects/java/target/cobertura").mkdirs();
360
361        LOGGER.info("   Start nodes");
362        StartNodesMojo startNodes = (StartNodesMojo) lookupConfiguredMojo(startPom, "start-nodes");
363        Assert.assertNotNull(startNodes);
364        startNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
365        simulatedLog.reset();
366        startNodes.setLog(simulatedLog);
367        startNodes.execute();
368        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
369        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
370        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog()
371            .contains("Node started"));
372
373        LOGGER.info("   Test nodes");
374        TestJavaFragmentMojo testMojo = (TestJavaFragmentMojo) lookupConfiguredMojo(testPom, "test-java-fragment");
375        Assert.assertNotNull(testMojo);
376        startNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
377        simulatedLog.reset();
378        testMojo.setLog(simulatedLog);
379        testMojo.optionsProperty = new String[]{"-verbose"};
380        testMojo.nodeOptionsProperty = new String[]{"debug=true", "ignoreoptionsfile=cwxxx"};
381        testMojo.execute();
382        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
383        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
384        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog()
385            .contains("Failures: 0, Errors: 0, Skipped: 0"));
386
387        LOGGER.info("   Test nodes with junit port");
388        startNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
389        simulatedLog.reset();
390        testMojo.setLog(simulatedLog);
391        testMojo.optionsProperty = new String[]{"-verbose"};
392        testMojo.nodeOptionsProperty = new String[]{"debug=true"};
393        testMojo.systemPropertyVariables = new HashMap<String, String>();
394        testMojo.systemPropertyVariables.put("com.tibco.junit.results.port", "2000");
395        testMojo.execute();
396
397        LOGGER.info("   Deploy nodes");
398        DeployFragmentMojo deployMojo = (DeployFragmentMojo) lookupConfiguredMojo(deployPom, "deploy-fragment");
399        Assert.assertNotNull(deployMojo);
400        startNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
401        simulatedLog.reset();
402        deployMojo.setLog(simulatedLog);
403        deployMojo.execute();
404        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
405        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
406
407        LOGGER.info("   Stop nodes");
408        StopNodesMojo stopNodes = (StopNodesMojo) lookupConfiguredMojo(stopPom, "stop-nodes");
409        Assert.assertNotNull(stopNodes);
410        simulatedLog.reset();
411        stopNodes.setLog(simulatedLog);
412        stopNodes.execute();
413        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
414        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
415        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog()
416            .contains("Node removed"));
417    }

420     * Test correct exit when non-daemon threads are still running.
421     *
422     * @throws Exception on error
423     */
424    @Test
425    public void testExitWhenThreadsRunning() throws Exception {
426
427        new Thread() {
428            public synchronized void run() {
429                try {
430                    wait();
431                } catch (final InterruptedException e) {
432                    LOGGER.warn("Interrupted", e);
433                }
434            }
435        }.start();
436    }
