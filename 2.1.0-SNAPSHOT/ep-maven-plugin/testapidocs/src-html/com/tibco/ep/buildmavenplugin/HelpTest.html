030package com.tibco.ep.buildmavenplugin;
031
032import java.io.File;
033
034import org.apache.maven.plugin.Mojo;
035import org.apache.maven.plugin.testing.MojoRule;
036import org.apache.maven.plugin.testing.resources.TestResources;
037import org.junit.Assert;
038import org.junit.Rule;
039import org.junit.Test;
040
041import org.slf4j.Logger;
042import org.slf4j.LoggerFactory;
043
044/**
045 * Unpack tests
046 *
047 */
048public class HelpTest extends BetterAbstractMojoTestCase  {
049    private Logger logger = LoggerFactory.getLogger(HelpTest.class);
050
051    /**
052     * rule
053     */
054    @Rule
055    public MojoRule rule = new MojoRule();
056
057    /**
058     * resources
059     */
060    @Rule
061    public TestResources resources = new TestResources();
062
063
064    /**
065     * Help
066     * 
067     * @throws Exception on error
068     */
069    @Test
070    public void testHelp() throws Exception {
071        SimulatedLog simulatedLog = new SimulatedLog(false);
072
073        File testPom = new File( "target/projects/eventflow", "pom.xml" );
074        Assert.assertNotNull(testPom);
075        Assert.assertTrue(testPom.exists());
076
077        Mojo help = lookupConfiguredMojo(testPom, "help");
078        Assert.assertNotNull(help);  
079        simulatedLog.reset();
080        help.setLog(simulatedLog);
081        help.execute();
082        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
083        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
084
085    }
086
087}
