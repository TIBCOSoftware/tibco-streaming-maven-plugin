119    private static List<Path> toUniqueAndExistingPaths(List<String> pathElements) throws MojoExecutionException {
120        LinkedHashSet<Path> paths = new LinkedHashSet<>(toPaths(pathElements));
121        return paths.stream()
122            .filter(p -> p.toFile().exists())
123            .collect(Collectors.toList());
124
125    }
126
127    private static List<Path> toPaths(List<String> stringPaths) throws MojoExecutionException {
128        List<Path> list = new ArrayList<>();
129        for (String p : stringPaths) {
130            list.add(toPath(p));
131        }
132        return list;
133    }

135    private static Path toPath(String file) throws MojoExecutionException {
136        return new File(file).toPath();
137    }

139    private File[] initializeAndCheck(File[] originalValue, String defaultDir) {
140
141        if (originalValue == null || originalValue.length == 0) {
142
143            //  Default value, if it exists.
144            //
145            File defaultValue = new File(project.getBasedir(), defaultDir);
146            if (defaultValue.exists() && defaultValue.isDirectory()) {
147                return new File[]{defaultValue};
148            } else {
149                return new File[]{};
150            }
151        }
152
153        return originalValue;
154    }

156    @Override
157    public void execute() throws MojoExecutionException, MojoFailureException {
158
159        if (skipGenerateSources) {
160            getLog().warn("Skipping code generation entirely");
161            return;
162        }
163
164        //  Transform the compiler properties into a map.
165        //
166        compilerProperties = new HashMap<>();
167        for (String prop : compilerPropertiesEntries) {
168            String[] propSplit = prop.split(COMPILER_PROPERTIES_EQUALS);
169            if (propSplit.length != 2) {
170                throw new MojoExecutionException("Illegal compiler property (not 'key=value'): "
171                    + prop);
172            }
173
174            compilerProperties.put(propSplit[0], propSplit[1]);
175        }
176
177        prechecks();
178        initializeService(PlatformService.CODE_GENERATION, ErrorHandling.FAIL);
179
180        eventflowDirectories = initializeAndCheck(eventflowDirectories, "/src/main/eventflow");
181        testEventflowDirectories = initializeAndCheck(testEventflowDirectories, "/src/test/eventflow");
182
183        setupEngineDataArea();
184
185        //  Construct the build parameters and trigger the build.
186        //
187        BuildParameters buildParameters = new BuildParameters();
188
189        try {
190            buildParameters
191                .withCompilerProperties(compilerProperties == null
192                    ? new HashMap<>() : compilerProperties)
193                .withSourcePaths(toPathsCheckExist(eventflowDirectories))
194                .withTestSourcePaths(toPathsCheckExist(testEventflowDirectories));
195
196            buildParameters
197                .withProjectCompileClassPath(
198                    toUniqueAndExistingPaths(project.getCompileClasspathElements()))
199                .withProjectTestCompileClassPath(
200                    toUniqueAndExistingPaths(project.getTestClasspathElements()));
201
202            buildParameters
203                .withDependenciesCompileClassPath(getDependencyClassPaths(BuildTarget.MAIN))
204                .withDependenciesTestCompileClassPath(getDependencyClassPaths(BuildTarget.TEST));
205
206            //  FIX THIS (FL): remove when SBRT has all parameters.
207            //
208            buildParameters
209                .withCompileClassPath(getCompileClassPath())
210                .withTestClassPath(getTestClassPath());
211
212            buildParameters
213                .withConfigurationDirectory(configurationDirectory.toPath())
214                .withTestConfigurationDirectory(testConfigurationDirectory.toPath())
215                .withBuildDirectory(toPath(project.getBuild().getDirectory()))
216                .withProductHome(productHome.toPath());
217
218        } catch (DependencyResolutionRequiredException e) {
219            throw new MojoExecutionException("Cannot resolve dependency", e);
220        }
221
222        logPaths("Compile ClassPath", buildParameters.getCompileClassPath());
223        logPaths("Test ClassPath", buildParameters.getTestClassPath());
224
225        logPaths("Project Compile ClassPath", buildParameters.getProjectCompileClassPath());
226        logPaths("Dependencies Compile ClassPath", buildParameters.getDependenciesCompileClassPath());
227        logPaths("Project Test Compile ClassPath", buildParameters.getProjectTestCompileClassPath());
228        logPaths("Dependencies Test Compile ClassPath", buildParameters.getDependenciesTestCompileClassPath());
229
230        //  Now trigger the build and report errors.
231        //
232        BuildNotifier notifier = new BuildNotifier();
233        try {
234
235            getBuildService()
236                .build(project.getName(), target, buildParameters, notifier);
237
238        } catch (FailFastException ffe) {
239
240            //  This is just used to get to failures quicker.
241            //
242            assert !failedBuilds.isEmpty();
243        }
244
245        if (!failedBuilds.isEmpty()) {
246            throw new MojoExecutionException(
247                "Code generation failed:\n(See above for stacks)\n"
248                    + String.join("\n", failedBuilds));
249        }
250
251        //  Now update the list of generated modules.
252        //  This list will be used to construct the manifest (so, we only do that for MAIN).
253        //
254        if (target == BuildTarget.MAIN) {
255            notifier.moduleData.write(project);
256        }
257
258        //  Add the generated source directory
259        //
260        addGeneratedSourceRoot();
261
262    }
263
264    private void logPaths(String header, List<Path> paths) {
265
266        if (paths.isEmpty()) {
267            getLog().debug(header + ": (none)");
268            return;
269        }
270
271        getLog().debug(header + ":");
272        paths.stream()
273            .map(Path::toString)
274            .sorted()
275            .forEach(p -> getLog().debug("  " + p));
276    }

278    private void setupEngineDataArea() throws MojoExecutionException {
279
280        //  This is used by ResourcePathResolvers to copy data into.
281        //
282        Path tempDirectory = Paths.get(project.getBuild().getDirectory()).resolve("tmp");
283        String tempDirectoryString = tempDirectory.toAbsolutePath().toFile().toString();
284        System.setProperty(ENGINE_DATA_AREA, tempDirectoryString);
285
286        //  Create the directory.
287        //
288        tempDirectory.toFile().mkdirs();
289        if (!tempDirectory.toFile().exists()) {
290            throw new MojoExecutionException("Could not create: " + tempDirectory);
291        }
292        getLog().debug(ENGINE_DATA_AREA + " set to " + tempDirectoryString);
293    }

295    private List<Path> getTestClassPath() throws MojoExecutionException, DependencyResolutionRequiredException {
296        //  FIX THIS (FL): not good since we need to add the CP of test dependencies.
297        //  To be finalized when getCompileClassPath() is ok.
298        //
299        return toPaths(project.getTestClasspathElements());
300    }
301
302    private List<Path> getCompileClassPath() throws MojoExecutionException, DependencyResolutionRequiredException {
303
304        //  Filter out compile class path elements that do not exist.
305        //
306        LinkedHashSet<Path> classpath = toPaths(project.getCompileClasspathElements()).stream()
307            .filter(p -> p.toFile().exists()).collect(Collectors.toCollection(LinkedHashSet::new));
308
309        //  Then dependencies.
310        //
311        visitDependencies((currentProjectArtefactDependency, indent, context) -> {
312
313            if (target == BuildTarget.MAIN
314                && currentProjectArtefactDependency.getScope().equals((Artifact.SCOPE_TEST))) {
315
316                getLog().debug(indent + currentProjectArtefactDependency
317                    + " [skipping (and skipping dependencies)]");
318                return false;
319            }
320
321            getLog().debug(indent + currentProjectArtefactDependency + " [adding]");
322
323            classpath.add(currentProjectArtefactDependency.getFile().toPath());
324
325            return true;
326        });
327
328        return new ArrayList<>(classpath);
329    }
330
331    private List<Path> getDependencyClassPaths(BuildTarget target) throws MojoExecutionException {
332        Set<Path> classpath = new LinkedHashSet<>();
333
334        getLog().debug("Visiting dependencies for target: " + target);
335        visitDependencies((currentProjectArtefactDependency, indent, context) -> {
336
337            if (this.target == BuildTarget.MAIN) {
338                //  If we're building MAIN, we don't want TEST scope dependencies.
339                //  But if we're building TEST, we want them all (no filtering).
340                //
341                if (currentProjectArtefactDependency.getScope().equals((Artifact.SCOPE_TEST))) {
342
343                    getLog().debug(indent + currentProjectArtefactDependency
344                        + " [skipping (and skipping dependencies): TEST scope]");
345                    return false;
346                }
347            }
348
349            getLog().debug(indent + currentProjectArtefactDependency + " [adding]");
350
351            classpath.add(currentProjectArtefactDependency.getFile().toPath());
352
353            return true;
354        });
355
356        return new ArrayList<>(classpath);
357    }

359    private void addGeneratedSourceRoot() {
360
361        //  FIX THIS (FL) The generated source path is hardcoded here and sb-server.
362        //  We should get this out of the build, possibly through the notifier.
363        //
364        String directory = project.getBuild().getDirectory() + "/generated-"
365            + (target == BuildTarget.MAIN ? "" : "test-")
366            + "sources/streaming";
367
368        if (target == BuildTarget.MAIN) {
369            project.addCompileSourceRoot(directory);
370        } else {
371            assert target == BuildTarget.TEST : target;
372            project.addTestCompileSourceRoot(directory);
373        }
374    }

377        return Stream.of(files)
378            .filter(File::exists)
379            .map(File::toPath).collect(Collectors.toList());
380    }

383        return Stream.of(files).map(File::toPath).collect(Collectors.toList());

386    private class BuildNotifier implements IBuildNotifier {
387
388        private final ProjectModuleData moduleData = new ProjectModuleData();
389
390        @Override
391        public void onBuildStarted(int nbModules) {
392            getLog().info("Found " + nbModules + " module" + (nbModules > 1 ? "s" : ""));
393        }
394
395        @Override
396        public void onBuildCompleted() {
397            //  Do nothing
398        }
399
400        @Override
401        public void onSkipped(String entityName, String extension) {
402            getLog().debug("Module " + entityName + ": code generation SKIPPED");
403            moduleData.addModule(entityName, extension);
404        }
405
406        @Override
407        public void onStarted(String entityName, String extension) {
408            getLog().debug("Module " + entityName + ": code generation STARTED");
409            moduleData.addModule(entityName, extension);
410        }
411
412        @Override
413        public void onCompleted(BuildResult result) {
414
415            double seconds = result.getElapsedTimeMillis();
416            seconds /= TimeUnit.MILLISECONDS.convert(1, TimeUnit.SECONDS);
417
418            String entityName = result.getEntityName();
419
420            if (!result.getException().isPresent()) {
421                getLog().debug("Module " + entityName
422                    + ": code generation SUCCESS"
423                    + " (in " + String.format("%.3f", seconds) + " seconds)");
424                return;
425            }
426
427            //  We have a failure.
428            //
429            getLog().error("Module " + entityName
430                + ": code generation FAILURE"
431                + " (in " + String.format("%.3f", seconds) + " seconds)");
432
433            Exception error = result.getException().get();
434            List<BuildExceptionDetails> detailList = getBuildService().getDetails(error);
435
436            assert !detailList.isEmpty() : entityName;
437            failedBuilds.add(entityName + ": " + detailList.get(0).getShortMessage());
438
439            int i = 0;
440            for (BuildExceptionDetails details : detailList) {
441
442                String header = "";
443                if (i > 0) {
444                    header += " [" + i + "] ";
445                }
446
447                if (details.getLocation() != null) {
448                    getLog().error(header + "Location: " + details.getLocation());
449                }
450
451                getLog().error(header + "Error: " + details.getShortMessage());
452
453                if (details.getLongDescription() != null) {
454                    getLog().error(
455                        header + "Detailed error: " + details.getLongDescription());
456                }
457
458                if (i == 0 && detailList.size() > 1) {
459                    getLog().error("Nested causes:");
460                }
461                i++;
462            }
463
464            getLog().error("Exception stack:");
465            List<String> causes = new ArrayList<>();
466            Throwable current = error;
467            while (current != null) {
468                causes.add("   At " + current.getStackTrace()[0].toString() + ": " + current
469                    .toString());
470                current = current.getCause();
471            }
472
473            Collections.reverse(causes);
474            causes.forEach(c -> getLog().error(c));
475
476            //  We don't want Maven to display a stack trace just because a build fail, so
477            //  we push the log with the exception as "DEBUG". Users can still get it with -X.
478            //
479            getLog().debug("Exception for above failure", error);
480
481            //  Stop on first error if needed.
482            //
483            if (failFast) {
484                throw new FailFastException();
485            }
486        }
487    }
