030package com.tibco.ep.buildmavenplugin;
031
032import java.io.File;
033import java.util.Map;
034import java.util.Properties;
035
036import org.apache.maven.model.PluginExecution;
037import org.apache.maven.plugin.MojoExecutionException;
038import org.apache.maven.plugins.annotations.Mojo;
039import org.apache.maven.plugins.annotations.Parameter;
040
041/**
042 * <p>Stop and remove nodes.</p>
043 * 
044 * <p>When executed in clean phase, any errors found will be logged but won't 
045 * cause the build to fail.  When executed in a phase other than clean, a stop 
046 * failure will fail the build.</p>
047 */
048@Mojo(name = "stop-nodes", threadSafe = true)
049public class StopNodesMojo extends BaseExecuteMojo {
050
051    /**
052     * <p>Set this to 'true' to skip stopping test nodes</p>
053     * 
054     * <p>Example use in pom.xml:</p>
055     * <img src="uml/start-nodes-skipStop.svg" alt="pom">
056     * 
057     * <p>Example use on commandline:</p>
058     * <img src="uml/start-nodes-skipStop-commandline.svg" alt="pom">
059     * 
060     * @since 1.0.0
061     */
062    @Parameter(property = "skipStop")
063    boolean skipStop;
064
065    /**
066     * <p>Set this to 'true' to skip removing test nodes</p>
067     * 
068     * <p>Example use in pom.xml:</p>
069     * <img src="uml/stop-nodes-skipRemove.svg" alt="pom">
070     * 
071     * <p>Example use on commandline:</p>
072     * <img src="uml/stop-nodes-skipRemove-commandline.svg" alt="pom">
073     * 
074     * @since 1.0.0
075     */
076    @Parameter(property = "skipRemove", defaultValue="false")
077    boolean skipRemove;
078    
079    /**
080     * Set this to 'false' to ignore errors on node stop
081     * 
082     * <p>Example use in pom.xml:</p>
083     * <img src="uml/stop-nodes-failOnStopError.svg" alt="pom">
084     * 
085     * <p>Example use on commandline:</p>
086     * <img src="uml/stop-nodes-failOnStopError-commandline.svg" alt="pom">
087     * 
088     * @since 1.0.0
089     */
090    @Parameter(property = "failOnStopError", defaultValue="true")
091    boolean failOnStopError;
092    
093    /**
094     * <p>List of node names.  If not set, a
095     * single node <b>A</b> is used.</p>
096     * 
097     * <p>Example use in pom.xml:</p>
098     * <img src="uml/start-nodes-nodes.svg" alt="pom">
099     * 
100     * @since 1.0.0
101     */
102    @Parameter
103    String[] nodes = new String[] {"A"};
104
105    /**
106     * Node installation path - overrides servicename, nodes are ignored
107     *
108     * @since 1.6.0
109     */
110    @Parameter
111    String installPath;
112    
113    // not used here so skip in docs
114    @Parameter(readonly = true)
115    BuldType buildtype;
116    @Parameter(readonly = true)
117    Map<String, String> environmentVariables;
118    @Parameter(readonly = true)
119    String[] ignoreLeaks;
120    @Parameter(property = "skipTests", readonly = true)
121    boolean skipTests;
122    @Parameter(readonly = true, property = "installOnly")
123    boolean installOnly;
124    
125    public void execute() throws MojoExecutionException {
126        getLog().debug("Stop nodes");
127
128        prechecks();
129
130        Properties modelProperties = project.getModel().getProperties();
131        boolean testCasesFound = true;
132        if (modelProperties.getProperty(TESTCASESFOUND_PROPERTY) != null && modelProperties.getProperty(TESTCASESFOUND_PROPERTY).equals("false")) {
133            testCasesFound = false;
134        }
135        
136        boolean clean = false;
137        if (execution.getLifecyclePhase() != null && (execution.getLifecyclePhase().equals("clean") || execution.getLifecyclePhase().equals("pre-clean"))) {
138            clean = true;
139        }
140        
141        // determine if we have executions steps in the pom - if so we can skip id this
142        // run is a default one
143        //
144        boolean hasExecutions = false;
145        if (mojoExecution.getPlugin() != null && mojoExecution.getPlugin().getExecutions() != null) {
146            for (PluginExecution p : mojoExecution.getPlugin().getExecutions()) {
147                if (!p.getId().startsWith("default-")) {
148                    hasExecutions = true;
149                }
150            }
151        }
152        
153        if (!clean && (
154                (hasExecutions && mojoExecution.getExecutionId().startsWith("default-")) || skipTests || installOnly || !testCasesFound)) {
155            getLog().info("Stop and remove nodes is skipped (skipTests="+skipTests+",hasExecutions="+hasExecutions+",testCasesFound="+testCasesFound+",installOnly="+installOnly+",phase="+execution.getLifecyclePhase()+")");
156            return;
157        }
158
159        ErrorHandling errorHandling = (failOnStopError ? ErrorHandling.FAIL : ErrorHandling.IGNORE);
160        if (clean) {
161            errorHandling = ErrorHandling.IGNORE;
162        }
163
164        if (!initializeService(PlatformService.ADMINISTRATION, errorHandling)) {
165
166            assert errorHandling == ErrorHandling.IGNORE;
167
168            //  We could not initialize.
169            //
170            return;
171        }
172
173        // shut down
174        //
175        if (skipStop) {
176            getLog().info("Stop nodes is skipped");
177        } else {
178            for (int i=0; i<nodes.length; i++) {
179                String nodeName = nodes[i]+"."+clusterName;
180                if (new File(nodeDirectory, nodeName).exists()) {
181                    stopNodes(nodeName, errorHandling);
182                }
183            }
184        }
185
186        if (skipRemove || skipStop) {
187            getLog().info("Remove nodes is skipped");
188            return;
189        }
190
191        // remove nodes
192        //
193        if (installPath != null) {
194            removeNode(installPath, errorHandling);
195        } else {
196            for (String node : nodes) {
197                String nodeName = node + "." + clusterName;
198                if (new File(nodeDirectory, nodeName).exists()) {
199                    try {
200                        removeNodes(nodeName, errorHandling);
201                    } catch (MojoExecutionException e) {
202                        removeNode(new File(nodeDirectory, nodeName)
203                            .getAbsolutePath(), errorHandling);
204                    }
205                }
206            }
207        }
208
209    }
210}
