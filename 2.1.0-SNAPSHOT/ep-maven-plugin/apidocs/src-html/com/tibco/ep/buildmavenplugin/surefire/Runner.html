030package com.tibco.ep.buildmavenplugin.surefire;
031
032import java.io.File;
033import java.util.ArrayList;
034import java.util.Arrays;
035import java.util.List;
036
037import org.apache.maven.surefire.Surefire;
038import org.apache.maven.surefire.report.BriefConsoleReporter;
039import org.apache.maven.surefire.report.BriefFileReporter;
040import org.apache.maven.surefire.report.ReporterException;
041import org.apache.maven.surefire.report.XMLReporter;
042import org.apache.maven.surefire.testset.TestSetFailedException;
043
044/**
045 * Test runner invoked on the server by the Deployment plugin.
046 */
047public class Runner
048{
049    
050    protected final static String SYS_PROP_PREFIX = "com.tibco.";
051    protected final static String SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER = SYS_PROP_PREFIX + "junit.results.";
052    
053    /**
054     * system property for reports
055     */
056    protected final static String REPORTS_DIRECTORY = "com.tibco.reportsdirectory";
057    
058    /**
059     * system property for node name - defined in the runtime but this would be an adanced definition
060     */
061    protected final static String NODE_NAME = "com.kabira.platform.node.name";
062    
063    /**
064     * cobertura sub directory
065     */
066    protected final static String COBERTURA_DIR = "cobertura";
067    
068    /**
069     * cobertura results file
070     */
071    protected final static String COBERTURA_RESULTS = "cobertura.ser";
072    
073    /**
074     * Main method to run a set of unit tests on the node.
075     * 
076     * @param args
077     *            a series of unit test class names
078     * @throws ReporterException Reporting error
079     * @throws TestSetFailedException  Test failure
080     */
081    public static void main(String[] args) throws ReporterException, TestSetFailedException {
082
083        // parse argumnets
084        //
085        // arg[0] "true" to use system exit on success, false otherwise
086        // args[1]+ contains junit test cases
087        //
088        boolean useSystemExit = Boolean.parseBoolean(args[0]);
089
090        List<Object[]> reportDefinition = new ArrayList<Object[]>();
091        List<Object> testSuites = new ArrayList<>();
092        Surefire surefire = new Surefire();
093
094        reportDefinition.add(new Object[] { BriefConsoleReporter.class.getName(), new Object[] { Boolean.TRUE } });
095        reportDefinition.add(new Object[] { XMLReporter.class.getName(), new Object[] { new File(System.getProperty(REPORTS_DIRECTORY)+File.separator+System.getProperty(NODE_NAME)), false } });
096        reportDefinition.add(new Object[] { BriefFileReporter.class.getName(), new Object[] { new File(System.getProperty(REPORTS_DIRECTORY)+File.separator+System.getProperty(NODE_NAME)), false } });
097
098        String[] testFileNames = null;
099        if (args.length > 1) {
100            testFileNames = Arrays.copyOfRange(args, 1, args.length);
101        }
102
103        // ================ (Optionally) Set up the Reporter that sends test results to the Studio-side JUnit view ================
104        //
105        // If the "SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER + port" System property is assigned a valid port number,
106        // then that means an EclipseJUnitViewSocketReporter instance should be created,
107        // for the sake of sending Surefire test results to the JUnit view in Studio.
108        //
109        // For the sake of development-level testing, the constructor accepts two debug-output System property booleans.
110        // When true, "SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER + tracecalls" prints to stdout whenever a Reporter interface call is made.
111        // When true, "SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER + tracesocketmessages" prints to stdout a copy of the messages sent to the
112        // JUnit client via the Socket connection.
113        //
114        Integer testResultsPort = 0;
115        String testResultsPortString = System.getProperty(SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER + "port");
116        if (testResultsPortString != null) {
117            try {
118                testResultsPort = Integer.parseInt(testResultsPortString);
119            } catch (NumberFormatException e) {
120                // fall-through
121            }
122        }
123        if (testResultsPort.intValue() != 0) {
124            Boolean traceCalls = Boolean
125                    .parseBoolean(System.getProperty(SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER + "tracecalls"));
126            Boolean traceSocketMessages = Boolean
127                    .parseBoolean(System.getProperty(SYS_PROP_PREFIX_JUNIT_RESULTS_REPORTER + "tracesocketmessages"));
128            reportDefinition.add(new Object[] { EclipseJUnitViewSocketReporter.class.getName(),
129                    new Object[] { testResultsPort, traceCalls, traceSocketMessages, testFileNames } });
130        }
131        // ========================================================================================================================
132
133
134        // force coverage to a node-specific file
135        //
136        System.setProperty("net.sourceforge.cobertura.datafile", System.getProperty(REPORTS_DIRECTORY)+File.separator+COBERTURA_DIR+File.separator+System.getProperty(NODE_NAME)+"."+COBERTURA_RESULTS);
137        File reportDirectory = new File(System.getProperty(REPORTS_DIRECTORY)+File.separator+COBERTURA_DIR);
138        if (!reportDirectory.exists()) {
139            // we could run this in locker but it seems safe to just ignore mkdirs if already created
140            reportDirectory.mkdirs();
141        }
142
143        if (args.length == 1) {
144            // FIX THIS - this shouldn't happen if called correctly from maven
145            //
146            String[] tclasses = new String[1];
147            testSuites.add(new Object[] { TestSuite.class.getName(), new Object[] { tclasses } });
148        } else {
149            testSuites.add(new Object[] { TestSuite.class.getName(), new Object[] { Arrays.copyOfRange(args, 1, args.length) } });
150        }
151
152        // FIX THIS: does not respect "failIfNoTests" parameter
153
154        ClassLoader.getSystemClassLoader().setDefaultAssertionStatus(true);
155
156        // Surefire....sigh. The run() method is not documented, but zero
157        // indicates SUCCESS. (The const is private so you'll have to trust me)
158        int retval = surefire.run(reportDefinition, testSuites, ClassLoader.getSystemClassLoader(),
159                ClassLoader.getSystemClassLoader(), Boolean.TRUE);
160
161        if (useSystemExit || retval != 0) {
162            System.exit(retval);
163        }
164    }
165}
