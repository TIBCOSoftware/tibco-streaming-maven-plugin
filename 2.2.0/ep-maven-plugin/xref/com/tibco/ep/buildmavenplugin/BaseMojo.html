240     /**
241      * Determine if the given artifact belongs to an EP platform.
242      * <p>
243      * This doesn't check for the artifact type, since there might be other
244      * attachments for that platform (sources, etc.). What it does mean
245      * is that if you've got a <code>zip</code> artifact for which this
246      * function returns <code>true</code>, it <em>can</em> be installed.
247      * By convention, that product-installable artifact has no classifier.
248      *
249      * @param artifact the artifact, with any relation to a platform
250      * @return whether the artifact belongs to an EP platform
251      */
252     static boolean isPlatformArtifact(Artifact artifact) {
253         final String artifactId = artifact.getArtifactId();
254         return artifactId.startsWith(SB_PRODUCT_ARTIFACT_PREFIX)
255             || artifactId.startsWith(SB_SUPPORT_ARTIFACT_PREFIX);
256     }

279     /**
280      * Initialize administration or build services
281      * <p>
282      * Admin & build jars are loaded
283      *
284      * @param service       The service to initialize
285      * @param errorHandling Error handling to indicate that errors must be reported not as exception
286      *                      but as a "false" return
287      * @return True if the initialization was successful. If errorHandling is {@code}
288      * @throws MojoExecutionException if initialize fails
289      */
290     boolean initializeService(PlatformService service, ErrorHandling errorHandling) throws MojoExecutionException {
291 
292         if (classLoader == null) {
293             AccessController.doPrivileged(new PrivilegedAction<Void>() {
294                 public Void run() {
295                     classLoader = new PluginClassloader(new URL[0], this.getClass()
296                         .getClassLoader());
297                     return null;
298                 }
299             });
300         }
301 
302         if (adminService == null && service == PlatformService.ADMINISTRATION) {
303 
304             loadJarForService(PlatformService.ADMINISTRATION);
305 
306             adminService = doGetService(
307                 () -> RuntimeServices.getAdminService(classLoader),
308                 IRuntimeAdminService.class,
309                 errorHandling);
310 
311             if (adminService != null) {
312 
313                 //  Only create a context if we have a service.
314                 //
315                 try {
316                     context = adminService.newContext(productHome.toPath());
317                 } catch (IllegalArgumentException e) {
318 
319                     if (errorHandling == ErrorHandling.IGNORE) {
320                         getLog().debug("Ignoring error", e);
321                         return false;
322                     }
323 
324                     throw new MojoExecutionException(
325                         "Could not instantiate context for product home " + productHome, e);
326                 }
327                 if (getLog().isDebugEnabled()) {
328                     context.withTracingEnabled();
329                 }
330             }
331         }
332 
333         if (buildService == null && service == PlatformService.CODE_GENERATION) {
334 
335             loadJarForService(PlatformService.CODE_GENERATION);
336 
337             buildService = doGetService(
338                 () -> RuntimeServices.getBuildService(classLoader),
339                 IRuntimeBuildService.class,
340                 errorHandling);
341         }
342 
343         return true;
344     }

346     private <T> T doGetService(Supplier<T> getter, Class<T> serviceClass, ErrorHandling errorHandling) throws MojoExecutionException {
347 
348         T service = getter.get();
349         if (service == null) {
350             String message = "Cannot get service " + serviceClass;
351 
352             if (errorHandling == ErrorHandling.FAIL) {
353                 throw new MojoExecutionException(message);
354             } else {
355                 getLog().warn(message + " [ignored]");
356             }
357         }
358 
359         return service;
360     }

362     /**
363      * to avoid maven dependency hell, we need to load the management jar and all its dependencies at
364      * runtime.
365      *
366      * @throws MojoExecutionException if jar loading fails
367      */
368     private void loadJarForService(PlatformService service) throws MojoExecutionException {
369 
370         // Needed for testing, since test stub API version is different from production version.
371         String productVersion = getProductVersion();
372 
373         if (productVersion.isEmpty()) {
374             throw new MojoExecutionException("Unable to determine the product version - ensure that either the management or sdk jar is set as a dependency in your pom");
375         }
376 
377         if (service == PlatformService.ADMINISTRATION) {
378             loadArtifact(DTM_GROUP_IDENTIFIER, DTM_MANAGEMENT_ARTIFACT_IDENTIFIER, productVersion);
379         } else {
380             assert service == PlatformService.CODE_GENERATION;
381             loadArtifact(SB_GROUP_IDENTIFIER, SB_SERVER_ARTIFACT_IDENTIFIER, productVersion);
382             loadArtifact(SB_RT_GROUP_IDENTIFIER, SB_CONTAINER_IDENTIFIER, productVersion);
383         }
384     }

386     private void loadArtifact(String groupIdentifier, String artifactIdentifier, String productVersion) throws MojoExecutionException {
387 
388         String jarCoordinates = groupIdentifier + ":" + artifactIdentifier + ":" + productVersion;
389         getLog().debug("Loading: " + jarCoordinates);
390 
391         Artifact managementAPIArtifact = repositorySystem
392             .createArtifact(groupIdentifier, artifactIdentifier, productVersion, "", "jar");
393         Set<Artifact> managementAPIDependencies = getProjectDependencies(
394             new HashSet<>(Collections.singletonList(managementAPIArtifact)));
395 
396         for (final Artifact artifact : managementAPIDependencies) {
397 
398             ArtifactResolutionRequest request = new ArtifactResolutionRequest();
399             request.setArtifact(artifact);
400             request.setRemoteRepositories(remoteRepositories);
401             request.setLocalRepository(localRepository);
402             repositorySystem.resolve(request);
403 
404             String artifactPath = getArtifactPath(artifact);
405             if (artifact.getGroupId().equals(groupIdentifier)
406                 && artifact.getArtifactId().equals(artifactIdentifier)) {
407                 getLog().debug("Loading: " + jarCoordinates
408                     + " from " + artifactPath + " (and dependencies)");
409             }
410 
411             try {
412 
413                 File file = new File(artifactPath);
414                 if (!file.exists()) {
415                     throw new FileNotFoundException("Failed to find:" + artifactPath);
416                 }
417 
418                 classLoader.addURL(file.toURI().toURL());
419 
420             } catch (Exception e) {
421                 throw new MojoExecutionException(
422                     "Unable to locate "
423                         + artifact.getGroupId()
424                         + "."
425                         + artifact.getArtifactId()
426                         + "-"
427                         + artifact.getVersion()
428                         + " (expected at "
429                         + artifactPath
430                         + ") - ensure that repository is on-line", e);
431             }
432         }
433     }

435     /**
436      * Return the absolute path to the given artifact.
437      *
438      * @param artifact Maven artifact to be resolved.
439      * @return Absolute path to the resolved artifact.
440      */
441     String getArtifactPath(Artifact artifact) {
442         assert artifact != null;
443 
444         if (artifact.getFile() != null) {
445 
446             // work-around for m2e getting the resolved path wrong
447             //
448             if (artifact.getFile().getName().equals("classes")) {
449                 String type = artifact.getType();
450                 if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type
451                     .equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
452                     File fixedPath = new File(artifact.getFile().getParentFile(), artifact
453                         .getArtifactId() + "-" + artifact.getVersion() + "-" + artifact
454                         .getType() + ".zip");
455                     artifact.setFile(fixedPath);
456                 }
457 
458                 if (type != null && (type.equals("nar"))) {
459                     File fixedPath = new File(artifact.getFile().getParentFile(), artifact
460                         .getArtifactId() + "-" + artifact.getVersion() + "-" + artifact
461                         .getClassifier() + ".nar");
462                     artifact.setFile(fixedPath);
463                 }
464             }
465 
466             return artifact.getFile().getAbsolutePath();
467         }
468 
469         // default to the local repository
470         //
471         String path = localRepository.getBasedir();
472         path += File.separator;
473         path += localRepository.pathOf(artifact);
474 
475         return path;
476     }

498     /**
499      * @param path  The path of a JAR
500      * @param entry The entry in the manifest to load
501      * @return The value from the manifest
502      */
503     Optional<String> getManifestEntry(String path, String entry) {
504 
505         try (JarInputStream jarStream = new JarInputStream(new FileInputStream(path))) {
506 
507             Manifest mf = jarStream.getManifest();
508             if (mf.getMainAttributes().containsKey(new Attributes.Name(entry))) {
509                 return Optional.of(mf.getMainAttributes().getValue(entry));
510             }
511 
512             return Optional.empty();
513 
514         } catch (IOException e) {
515             getLog().warn("No manifest could be read for: " + path, e);
516             return Optional.empty();
517         }
518     }

520     /**
521      * get the project dependencies as artifacts
522      *
523      * @return set of artifacts
524      * @throws MojoExecutionException on error
525      */
526     Set<Artifact> getCompileProjectDependencies() throws MojoExecutionException {
527 
528         Set<Artifact> compileDependencies = new LinkedHashSet<>();
529         Map<String, String> fragmentDependencies = new HashMap<>();
530 
531         visitDependencies(((artifact, indent, context) -> {
532 
533             String scope = artifact.getScope();
534             if (scope.equals(Artifact.SCOPE_COMPILE) || scope.equals(Artifact.SCOPE_RUNTIME)) {
535 
536                 //  Keep these.
537                 //
538                 compileDependencies.add(artifact);
539 
540                 if (isFragment(artifact)) {
541 
542                     getLog().debug(indent + artifact + " [adding (skip nested dependencies)]");
543 
544                     //  Keep track of this artifact dependencies to avoid duplicates.
545                     //
546                     getManifestEntry(getArtifactPath(artifact), MANIFEST_TIBCO_EP_FRAGMENT_LIST)
547                         .ifPresent(fragmentList -> {
548                             for (String fragmentDep : fragmentList.split(" ")) {
549                                 fragmentDependencies.put(fragmentDep, artifact.toString());
550                             }
551                         });
552 
553 
554                     //  Don't scan fragment dependencies.
555                     //
556                     return false;
557 
558                 }
559 
560                 //  Standard dependency (not a fragment), keep scanning.
561                 //
562                 getLog().debug(indent + artifact + " [adding]");
563                 return true;
564 
565             } else if (scope.equals(Artifact.SCOPE_PROVIDED) || scope.equals(Artifact.SCOPE_TEST)) {
566 
567                 //  Don't add these.
568                 //
569                 getLog().debug(indent + artifact + " [skipping (and skipping dependencies)]");
570                 return false;
571 
572             } else {
573 
574                 //  Skip, but visit dependencies.
575                 //
576                 getLog().debug(indent + artifact + "[skipping]");
577                 return true;
578             }
579         }));
580 
581         //  Remove fragments nested in other fragments.
582         //  FIX THIS (FL): remove nesting.
583         //
584         for (Iterator<Artifact> iterator = compileDependencies.iterator(); iterator.hasNext(); ) {
585 
586             Artifact artifact = iterator.next();
587             String destName = artifact.getGroupId() + "-" + artifact
588                 .getArtifactId() + "-" + artifact.getBaseVersion() + "-" + artifact
589                 .getType() + ".zip";
590 
591             if (fragmentDependencies.containsKey(destName)) {
592                 getLog()
593                     .debug("Skipping " + artifact + " since it is already included by "
594                         + fragmentDependencies.get(destName));
595                 iterator.remove();
596             }
597         }
598 
599         return compileDependencies;
600     }

602     /**
603      * Visit dependencies.
604      *
605      * @param visitor The visitor
606      * @throws MojoExecutionException Something bad happened
607      */
608     void visitDependencies(DependencyVisitor visitor) throws MojoExecutionException {
609         DependencyVisitorContext dependencyVisitorContext = new DependencyVisitorContext(visitor);
610 
611         ProjectBuildingRequest buildingRequest = new DefaultProjectBuildingRequest(session
612             .getProjectBuildingRequest());
613         buildingRequest.setProject(project);
614 
615         try {
616             DependencyNode rootNode = dependencyGraphBuilder
617                 .buildDependencyGraph(buildingRequest, null, reactorProjects);
618             rootNode.accept(dependencyVisitorContext);
619         } catch (DependencyGraphBuilderException e) {
620             throw new MojoExecutionException(e.getMessage(), e);
621         }
622 
623         if (dependencyVisitorContext.getException() != null) {
624             throw dependencyVisitorContext.getException();
625         }
626     }

696     /**
697      * get the project dependencies as artifacts, but limit scope to a subset
698      *
699      * @param artifacts starting artifacts
700      * @return set of artifacts
701      */
702     Set<Artifact> getProjectDependencies(Set<Artifact> artifacts) {
703 
704         final ArtifactFilter filter = artifact -> true;
705 
706         // note that resolving transitively here incorrectly brings in older dependencies
707         //
708         ArtifactResolutionRequest request = new ArtifactResolutionRequest();
709         request.setLocalRepository(localRepository);
710         request.setResolveRoot(false);
711         request.setRemoteRepositories(project.getRemoteArtifactRepositories());
712         request.setResolveTransitively(false);
713         request.setArtifact(project.getArtifact());
714         request.setArtifactDependencies(artifacts);
715         request.setManagedVersionMap(project.getManagedVersionMap());
716         request.setCollectionFilter(filter);
717         request.setOffline(session.isOffline());
718         request.setForceUpdate(session.getRequest().isUpdateSnapshots());
719         request.setServers(session.getRequest().getServers());

794     /**
795      * Check environment before running mojo
796      */
797     public void prechecks() {
798 
799         if (productHome == null || productHome.getAbsolutePath().isEmpty()) {
800             String epHome = System.getenv("TIBCO_EP_HOME");
801             if (epHome != null) {
802                 productHome = new File(epHome);
803             } else {
804 
805                 File base = new File(session.getLocalRepository().getBasedir()).getParentFile();
806 
807                 // locate product version
808                 //
809                 String baseSubDirectory = "";
810                 for (Artifact artifact : getProjectDependencies("zip")) {
811 
812                     if (isSBProduct(artifact)) {
813                         baseSubDirectory = artifact.getGroupId() + File.separator + artifact
814                             .getArtifactId() + File.separator + artifact.getBaseVersion();
815                         break;
816                     }
817                     if (isDTMProduct(artifact)) {
818                         baseSubDirectory = artifact.getGroupId() + File.separator + artifact
819                             .getArtifactId() + File.separator + artifact.getBaseVersion();
820                         break;
821                     }
822                 }
823                 productHome = new File(base, baseSubDirectory);
824 
825             }
826         }
827         getLog().debug("Product home set to " + productHome);
828     }

964     /**
965      * The dependency visitor context
966      */
967     class DependencyVisitorContext implements DependencyNodeVisitor {
968 
969         private final DependencyVisitor visitor;
970         private boolean logged = false;
971         private int depth = 0;
972         private MojoExecutionException exception;
973 
974         private DependencyVisitorContext(DependencyVisitor visitor) {
975             this.visitor = visitor;
976         }
977 
978         private MojoExecutionException getException() {
979             return exception;
980         }
981 
982         /**
983          * @param exception Set an exception that happened during {@link #visit(DependencyNode)}
984          */
985         void setException(MojoExecutionException exception) {
986             assert exception != null;
987             this.exception = exception;
988         }
989 
990         @Override
991         public boolean visit(DependencyNode node) {
992 
993             if (exception != null) {
994                 //  Just end quickly.
995                 return false;
996             }
997 
998             if (!logged) {
999                 getLog().debug("Processing dependency tree:");
1000                 logged = true;
1001             }
1002 
1003             if (node.getParent() == null) {
1004                 // root node - skip
1005                 return true;
1006             }
1007 
1008             StringBuilder t = new StringBuilder();
1009             for (int i = 0; i < depth; i++) {
1010                 t.append("  ");
1011             }
1012             String indent = t.toString();
1013             depth += 1;
1014 
1015             return visitor.visit(node.getArtifact(), indent, this);
1016         }
1017 
1018         @Override
1019         public boolean endVisit(DependencyNode node) {
1020 
1021             if (node.getParent() != null) {
1022                 depth = depth - 1;
1023             }
1024             return true;
1025         }
1026     }
