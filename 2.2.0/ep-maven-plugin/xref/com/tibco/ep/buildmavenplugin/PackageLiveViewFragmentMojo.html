30  package com.tibco.ep.buildmavenplugin;
31  
32  import org.apache.maven.plugin.MojoExecutionException;
33  import org.apache.maven.plugins.annotations.Mojo;
34  import org.apache.maven.plugins.annotations.Parameter;
35  import org.apache.maven.plugins.annotations.ResolutionScope;
36  import org.apache.maven.plugins.assembly.model.FileSet;
37  
38  import java.io.File;
39  
40  import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
41  
42  /**
43   * <p>Build a LiveView fragment</p>
44   *
45   * <p>The packaging rules are as follows :-</p>
46   * <ol>
47   * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
48   *      <ul>
49   *              <li>Archiver-Version: Plexus Archiver</li>
50   *              <li>Built-By: build user</li>
51   *              <li>Build-Jdk: jdk version</li>
52   *              <li>Package-Title: project groupId and name</li>
53   *              <li>Package-Version: project version and buildNumber</li>
54   *              <li>Package-Vendor: project organization name (if set)</li>
55   *              <li>TIBCO-EP-Build-Product-Version: product version</li>
56   *      </ul>
57   * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
58   * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
59   * <li>The project's files are copied into /</li>
60   * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
61   * <li>The project's resources files are copied into /</li>
62   * </ol>
63   *
64   * <p>The plexus archiver is used to create the archive via the maven assembly
65   * plugin.</p>
66   *
67   * <p>Native archives (nar) are included - in this case a mapping is included
68   * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
69   * values.</p>
70   *
71   * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-liveview-fragment.zip</p>
72   */
73  @Mojo(name = "package-liveview-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true)
74  public class PackageLiveViewFragmentMojo extends BasePackageMojo {
75  
76      /**
77       * <p>Eventflow source directories</p>
78       *
79       * <p>If no eventflowDirectories is specified, a single directory of
80       * ${project.basedir}/src/main/eventflow is used.</p>
81       *
82       * <p>Example use in pom.xml:</p>
83       * <img src="uml/eventflowDirectories.svg" alt="pom">
84       *
85       * <p>Example use on commandline:</p>
86       * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
87       *
88       * @since 1.0.0
89       */
90      @Parameter(required = false, property = "eventflowDirectories")
91      File[] eventflowDirectories;
92  
93      /**
94       * <p>Liveview source directory</p>
95       *
96       * <p>Example use in pom.xml:</p>
97       * <img src="uml/liveviewDirectory.svg" alt="pom">
98       *
99       * <p>Example use on commandline:</p>
100      * <img src="uml/liveviewDirectory-commandline.svg" alt="pom">
101      *
102      * @since 1.0.0
103      */
104     @Parameter(defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory")
105     File liveviewDirectory;
106 
107     @Override
108     public void execute() throws MojoExecutionException {
109         getLog().debug("Creating liveview fragment");
110 
111         eventflowDirectories = getOrDefaultSrcMainEventflow(eventflowDirectories);
112 
113         prechecks();
114 
115         newArchiveGenerator()
116             .withAssemblyStep(assembly -> {
117 
118                 //  Add files in our source directories.
119                 //
120                 for (File eventflowDirectory : eventflowDirectories) {
121                     FileSet fileSet = new FileSet();
122                     fileSet.setDirectory(eventflowDirectory.getAbsolutePath());
123                     fileSet.setOutputDirectory("");
124                     assembly.addFileSet(fileSet);
125                 }
126 
127                 FileSet fileSet = new FileSet();
128                 fileSet.setDirectory(liveviewDirectory.getAbsolutePath());
129                 fileSet.setOutputDirectory("");
130                 assembly.addFileSet(fileSet);
131 
132             })
133             .generate();
134     }
135 }
