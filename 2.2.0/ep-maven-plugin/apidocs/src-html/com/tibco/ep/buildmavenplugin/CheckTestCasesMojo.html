030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.MojoExecutionException;
033import org.apache.maven.plugins.annotations.Mojo;
034import org.apache.maven.plugins.annotations.Parameter;
035
036import static org.apache.maven.plugins.annotations.LifecyclePhase.TEST;
037
038import java.util.Properties;
039
040/**
041 * <p>Check for test cases.</p>
042 * 
043 * <p>If no test cases are found, <b>testCasesFound</b> property is set to true so that nodes are
044 * not started or stopped.</p>
045 */
046@Mojo(name = "check-testcases", defaultPhase = TEST, threadSafe = true)
047public class CheckTestCasesMojo extends BaseTestMojo {
048
049    /**
050     * <p>Set this to 'true' to skip running tests, but still compile them.</p>
051     * 
052     * <p>Example use in pom.xml:</p>
053     * <img src="uml/start-nodes-skipTests.svg" alt="pom">
054     * 
055     * <p>Example use on commandline:</p>
056     * <img src="uml/start-nodes-skipTests-commandline.svg" alt="pom">
057     * 
058     * @since 1.0.0
059     */
060    @Parameter(property = "skipTests", defaultValue = "false")
061    boolean skipTests;
062    
063    public void execute() throws MojoExecutionException {
064
065        getLog().debug( "Checking for test cases");
066
067        if (skipTests || installOnly) {
068            getLog().info("Tests are skipped (skipTests="+skipTests+",installOnly="+installOnly+")");
069            return;
070        }
071        
072        Properties modelProperties = project.getModel().getProperties();
073        
074        String[] testClasses = scanForTests();
075        if (testClasses == null || testClasses.length == 0) {
076            getLog().info("No test cases found, setting testCasesFound=false to skip redundant start and stop nodes");
077            modelProperties.setProperty(TESTCASESFOUND_PROPERTY, "false");
078        } else {
079            modelProperties.setProperty(TESTCASESFOUND_PROPERTY, "true");
080        }
081        
082        project.getModel().setProperties(modelProperties);
083
084    }
085
086}
