031package com.tibco.ep.buildmavenplugin.admin;
032
033import com.tibco.ep.buildmavenplugin.ErrorHandling;
034import com.tibco.ep.sb.services.management.AbstractCommandBuilder;
035import com.tibco.ep.sb.services.management.ICommand;
036import org.apache.maven.plugin.MojoExecutionException;
037import org.apache.maven.plugin.logging.Log;
038
039import java.util.HashMap;
040import java.util.Map;
041import java.util.Optional;
042import java.util.function.Consumer;
043
044/**
045 * An administrative command runner
046 */
047public class RuntimeCommandRunner {
048
049    private final Log log;
050    private final AbstractCommandBuilder builder;
051    private final String shortLocation;
052    private final String longLocation;
053
054    private boolean wait;
055    private ErrorHandling errorHandling;
056    private boolean recordOutput;
057    private Optional<Consumer<ICommand>> shutdownHook;
058    private ErrorHandler errorHandler;
059
060    private boolean shuttingDown;
061    private Thread shutdownHookThread;
062
063    /**
064     * @param log           The Maven log handle
065     * @param builder       The command builder
066     * @param shortLocation Short location for logging
067     * @param longLocation  Long location for error reporting
068     */
069    public RuntimeCommandRunner(Log log, AbstractCommandBuilder builder, String shortLocation, String longLocation) {
070        this.log = log;
071        this.builder = builder;
072        this.shortLocation = shortLocation;
073        this.longLocation = longLocation;
074
075        wait = true;
076        errorHandling = ErrorHandling.FAIL;
077        recordOutput = false;
078        shutdownHook = Optional.empty();
079        errorHandler = (runner, resultCode) -> {
080            throw new MojoExecutionException(
081                "Command: " + runner.builder.getCommand() + " " + runner.builder.getTarget()
082                    + " failed: " + runner.longLocation + " error code " + resultCode);
083        };
084
085        shutdownHookThread = null;
086        shuttingDown = false;
087    }
088
089    /**
090     * @return The command description (command + target)
091     */
092    private String getDescription() {
093        return builder.getCommand() + " " + builder.getTarget();
094    }
095
096    /**
097     * Should the runner wait for command completion ?
098     *
099     * @param wait True if the runner needs to wait
100     * @return This
101     */
102    public RuntimeCommandRunner wait(boolean wait) {
103        this.wait = wait;
104        return this;
105    }
106
107    /**
108     * Should the runner fail upon error ?
109     *
110     * @param errorHandling The error handling for the command
111     * @return This
112     */
113    public RuntimeCommandRunner errorHandling(ErrorHandling errorHandling) {
114        this.errorHandling = errorHandling;
115        return this;
116    }
117
118    /**
119     * Should the runner record output
120     *
121     * @param recordOutput True if the runner should record output
122     * @return This
123     */
124    public RuntimeCommandRunner recordOutput(boolean recordOutput) {
125        this.recordOutput = recordOutput;
126        return this;
127    }
128
129    /**
130     * Configure a shutdown hook
131     *
132     * @param shutdownHook The shutdown hook
133     * @return This
134     */
135    public RuntimeCommandRunner shutdownHook(Consumer<ICommand> shutdownHook) {
136        this.shutdownHook = Optional.of(shutdownHook);
137        return this;
138    }
139
140    /**
141     * Configure an error hook
142     *
143     * @param errorHandler The error hook
144     * @return This
145     */
146    public RuntimeCommandRunner onError(ErrorHandler errorHandler) {
147        this.errorHandler = errorHandler;
148        return this;
149    }
150
151    /**
152     * Run the command
153     *
154     * @return This
155     * @throws MojoExecutionException An execution failure happened
156     */
157    public RuntimeCommandNotifier run() throws MojoExecutionException {
158        return run(new HashMap<>());
159    }
160
161    /**
162     * Run the command with parameters
163     *
164     * @param parameters The parameters
165     * @return This
166     * @throws MojoExecutionException An execution failure happened
167     */
168    public RuntimeCommandNotifier run(Map<String, String> parameters) throws MojoExecutionException {
169
170        try {
171            ICommand command = builder.build();
172
173            //  Install shutdown hook if needed.
174            //
175            if (shutdownHook.isPresent()) {
176
177                //  We will always fail on error if we have a shutdown hook to cleanup failures.
178                //
179                assert errorHandling == ErrorHandling.FAIL;
180
181                shutdownHookThread = new Thread(() -> {
182                    shuttingDown = true;
183                    shutdownHook.get().accept(command);
184                });
185                Runtime.getRuntime().addShutdownHook(shutdownHookThread);
186            }
187
188            RuntimeCommandNotifier notifier = new RuntimeCommandNotifier(log, getDescription(), shortLocation);
189            notifier.recordOutput(recordOutput).errorHandling(errorHandling);
190
191            command.execute(parameters, notifier);
192
193            if (!wait) {
194                return notifier;
195            }
196
197            int resultCode = command.waitForCompletion();
198
199            if (shuttingDown) {
200                assert shutdownHook.isPresent();
201                throw new MojoExecutionException(
202                    "command: " + getDescription() + " was aborted - attempting to clean up");
203            }
204
205            if (resultCode != 0 && errorHandling == ErrorHandling.FAIL) {
206                errorHandler.onError(this, resultCode);
207            }
208
209            removeShutdownHook();
210
211            return notifier;
212
213        } catch (IllegalArgumentException iae) {
214
215            removeShutdownHook();
216
217            throw new MojoExecutionException(
218                "Illegal argument while using administrative API, command: "
219                    + getDescription() + " on " + longLocation, iae);
220        }
221    }
222
223    /**
224     * Remove the shutdown hook
225     */
226    public void removeShutdownHook() {
227        if (shutdownHookThread != null) {
228            Runtime.getRuntime().removeShutdownHook(shutdownHookThread);
229        }
230        shutdownHookThread = null;
231    }
232}
