031package com.tibco.ep.sb.services.stubs.admin;
032
033import com.tibco.ep.sb.services.management.AbstractDeployFragmentCommandBuilder;
034import com.tibco.ep.sb.services.management.FragmentType;
035import com.tibco.ep.sb.services.management.IDeployFragmentCommand;
036import com.tibco.ep.sb.services.management.IDestination;
037import com.tibco.ep.sb.services.management.INotifier;
038
039import java.util.Arrays;
040import java.util.List;
041import java.util.Map;
042
043/**
044 * The deploy fragment stub implementation
045 */
046public class DeployFragmentCommand extends Command implements IDeployFragmentCommand {
047
048    private final boolean fail;
049
050    /**
051     * @param builder The builder
052     */
053    public DeployFragmentCommand(AbstractDeployFragmentCommandBuilder builder) {
054        super(builder);
055
056        List<String> options = builder.getExecutionOptions();
057
058        long count = 0;
059        for (String s : options) {
060            if (s.startsWith("ignoreoptionsfile")) {
061                count++;
062            }
063        }
064
065        this.fail = (count > 1);
066
067        //  Setup test completion.
068        //
069        for (String target : Arrays.asList(
070            "com.tibco.ep.buildmavenplugin.surefire.Runner",
071            "com.tibco.ep.buildmavenplugin.DummyTest")) {
072
073            upon("deploy", target,
074                (params, notifier) -> {
075                    notifier.info("", "Failures: 0, Errors: 0, Skipped: 0");
076                });
077        }
078    }
079
080    @Override
081    public void execute(Map<String, String> parameters, INotifier notifier) {
082        if (fail) {
083            throw new IllegalStateException("Invalid parameters");
084        }
085        super.execute(parameters, notifier);
086    }
087
088    /**
089     * The builder class
090     */
091    static class Builder extends AbstractDeployFragmentCommandBuilder {
092
093        /**
094         * @param destination  The destination
095         * @param fragmentType The fragment type
096         * @param target       The target
097         */
098        public Builder(IDestination destination, FragmentType fragmentType, String target) {
099            super(destination, fragmentType, target);
100        }
101
102        @Override
103        public IDeployFragmentCommand build() {
104            return new DeployFragmentCommand(this);
105        }
106    }
107}
