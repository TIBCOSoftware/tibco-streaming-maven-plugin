030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.MojoExecutionException;
033import org.apache.maven.plugins.annotations.LifecyclePhase;
034import org.apache.maven.plugins.annotations.Mojo;
035import org.apache.maven.plugins.annotations.Parameter;
036
037import java.io.File;
038import java.io.IOException;
039import java.nio.file.FileVisitResult;
040import java.nio.file.Files;
041import java.nio.file.Path;
042import java.nio.file.Paths;
043import java.nio.file.SimpleFileVisitor;
044import java.nio.file.StandardCopyOption;
045import java.nio.file.attribute.BasicFileAttributes;
046
047/**
048 * <p>Compile eventflow</p>
049 * 
050 * <p>Compilation includes copying the source files to the output directory</p>
051 */
052@Mojo(name = "compile-eventflow-fragment", defaultPhase = LifecyclePhase.COMPILE, threadSafe = true)
053public class CompileEventFlow extends BaseMojo {
054
055    /**
056     * <p>Eventflow source directories</p>
057     * 
058     * <p>If no eventflowDirectories is specified, a single directory of
059     * ${project.basedir}/src/main/eventflow is used.</p>
060     * 
061     * <p>Example use in pom.xml:</p>
062     * <img src="uml/eventflowDirectories.svg" alt="pom">
063     * 
064     * <p>Example use on commandline:</p>
065     * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
066     * 
067     * @since 1.0.0
068     */
069    @Parameter( required = false, property = "eventflowDirectories" )
070    File[] eventflowDirectories;
071
072    public void execute() throws MojoExecutionException {       
073        
074        getLog().debug( "Compiling streambase fragment");
075
076        eventflowDirectories = getOrDefaultEventFlowDirectories(eventflowDirectories);
077
078        prechecks();
079
080        for (File eventflowDirectory : eventflowDirectories) {
081            if (eventflowDirectory.exists()) {
082                try {
083                    Files.walkFileTree(eventflowDirectory.toPath(), new SimpleFileVisitor<Path>() {
084
085                        @Override
086                        public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
087                            try {
088                                Files.createDirectories(Paths.get(project.getBuild().getOutputDirectory(), eventflowDirectory.toPath().relativize(dir).toString()));
089                            } catch (IOException e) {
090                                // ignore
091                            }
092                            return FileVisitResult.CONTINUE;
093                        }
094
095                        @Override
096                        public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
097                            if (file.toString().endsWith(".sbapp") || file.toString().endsWith(".sbint") || file.toString().endsWith(".ssql")) {
098
099                                // copy file to output directory
100                                //                                
101                                Files.copy(file, Paths.get(project.getBuild().getOutputDirectory(), eventflowDirectory.toPath().relativize(file).toString()), StandardCopyOption.REPLACE_EXISTING);
102
103                                // FIX THIS - needs to be re-visted once pacakging has settled dow
104
105                                /*
106                                if (file.toString().endsWith(".sbapp")) {
107                                    File compiled = compileSbapp(file.getFileName().toString());
108
109                                    if (compiled != null) {
110                                        FileItem compiledFile = new FileItem();
111                                        compiledFile.setOutputDirectory("dependencies"+File.separator+"java");
112                                        compiledFile.setSource(compiled.toString());
113                                        compiledFile.setDestName(file.getFileName()+"-compiled.jar");
114                                    }
115                                }
116                                 */
117
118                            }
119                            return FileVisitResult.CONTINUE;
120                        }
121                    });
122                } catch (IOException e) {
123                    throw new MojoExecutionException("Failed to read resource directory: " + e.getMessage(), e);
124                }
125            }
126        }
127    }
128
129}
