44  /**
45   * An object representing the module data: the list of interfaces and modules.
46   */
47  public class ProjectModuleData {
48  
49      //  The file containing the list of modules.
50      //
51      private static final String MODULES_FILE = "modules";

53      private static final String SBAPP_EXTENSION = ".sbapp";
54      private static final String SBINT_EXTENSION = ".sbint";

56      private final StringBuilder modules = new StringBuilder();
57      private final StringBuilder interfaces = new StringBuilder();
58      private boolean modulesIsEmpty = true;
59      private boolean interfacesIsEmpty = true;

65       */

67          File modulesFile = Paths.get(project.getBuild().getDirectory(), MODULES_FILE).toFile();

81      /**
82       * Add a module
83       *
84       * @param entityTypeName The module name
85       * @param extension      The extension
86       */
87      void addModule(String entityTypeName, String extension) {
88  
89          if (SBINT_EXTENSION.equals(extension)) {
90  
91              if (!interfacesIsEmpty) {
92                  interfaces.append(" ");
93              }
94              interfaces.append(entityTypeName);
95              interfacesIsEmpty = false;
96              return;
97          }
98  
99          assert SBAPP_EXTENSION.equals(extension) : extension + " for " + entityTypeName;
100         if (!modulesIsEmpty) {
101             modules.append(" ");
102         }
103         modules.append(entityTypeName);
104         modulesIsEmpty = false;
105     }
106 
107     /**
108      * Write the data to the filesystem.
109      *
110      * @param project The project
111      * @throws MojoExecutionException Could not write the file
112      */
113     void write(MavenProject project) throws MojoExecutionException {
114         File modulesFile = Paths.get(project.getBuild().getDirectory(), MODULES_FILE).toFile();
115 
116         try (BufferedWriter writer = new BufferedWriter(new FileWriter(modulesFile))) {
117 
118             writer.write(modules + "\n");
119             writer.write(interfaces + "\n");
120 
121         } catch (IOException exception) {
122             throw new MojoExecutionException("Couldn't write to file: " + modulesFile, exception);
123         }
124     }
125 
126     /**
127      * @return The space separated module list
128      */
129     public String getModules() {
130         return modules.toString();
131     }
132 
133     /**
134      * @return The space separated interface list
135      */
136     public String getInterfaces() {
137         return interfaces.toString();
138     }
