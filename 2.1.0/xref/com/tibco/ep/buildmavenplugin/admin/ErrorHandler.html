31  package com.tibco.ep.buildmavenplugin.admin;
32  
33  import org.apache.maven.plugin.MojoExecutionException;
34  
35  /**
36   * A functional interface to specify behavior upon error in a command execution
37   */
38  @FunctionalInterface
39  public interface ErrorHandler {
40      /**
41       * @param runner     The runner
42       * @param resultCode The result code
43       * @throws MojoExecutionException To report errors to Maven
44       */
45      void onError(RuntimeCommandRunner runner, int resultCode) throws MojoExecutionException;
46  }
