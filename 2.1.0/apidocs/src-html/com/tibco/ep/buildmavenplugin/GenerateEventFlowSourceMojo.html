031package com.tibco.ep.buildmavenplugin;
032
033import com.tibco.ep.sb.services.build.BuildTarget;
034import org.apache.maven.plugins.annotations.Mojo;
035
036import static org.apache.maven.plugins.annotations.LifecyclePhase.GENERATE_SOURCES;
037
038
039/**
040 * <p>Typecheck EventFlows and generate java sources</p>
041 *
042 * <p>The plugin will scan the configured EventFlow source directories and typecheck/build them in the
043 * dependency order, reporting any typecheck failure.</p>
044 *
045 * <p>It does not build the generated java sources, this is done using a standard maven compiler
046 * plugin execution.</p>
047 *
048 * <p>The plugin keeps track of individual SBAPP/SBINT dependencies and will trigger or skip a
049 * a rebuild accordingly.</p>
050 *
051 * <p>SBAPPs and SBINTs have public interfaces and private implementation details. The public
052 * interface is composed of anything that another SBAPP/SBINT can import. Schemas, Streams, QueryTable,
053 * Constants are part of the public interface. Everything else (Module operators, etc) is private.</p>
054 *
055 * <p>If an EventFlow A depends on another EventFlow B and A and B are in the same fragment, then,
056 * upon a change on B's public interface, the plugin will rebuild B, then A. Upon a change on B's
057 * implementation only, the plugin will only rebuild B.</p>
058 *
059 * <p>If an EventFlow A depends on another EventFlow B and A and B are in different fragments, the
060 * plugin will only trigger a rebuild of A if the rebuild of B has happened and is installed in the
061 * maven repository (i.e. if the fragment containing B has been rebuilt successfully).</p>
062 *
063 * <p>EventFlow builds are shared between Studio and the Maven command line. A fragment built using
064 * the command line will not get rebuilt by Studio and vice-versa. A Maven clean or the removal of
065 * the default *target* directory will force a rebuild on the next call.</p>
066 *
067 */
068@Mojo(name = "generate-main-eventflow", defaultPhase = GENERATE_SOURCES, threadSafe = false)
069public class GenerateEventFlowSourceMojo extends BaseGenerateMojo {
070
071    /**
072     * Public constructor
073     */
074    public GenerateEventFlowSourceMojo() {
075        super(BuildTarget.MAIN);
076    }
077}
