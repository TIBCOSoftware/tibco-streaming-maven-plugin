030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.MojoExecutionException;
033import org.apache.maven.plugins.annotations.Mojo;
034import org.apache.maven.plugins.annotations.Parameter;
035import org.apache.maven.plugins.annotations.ResolutionScope;
036import org.apache.maven.plugins.assembly.model.FileSet;
037
038import java.io.File;
039
040import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
041
042/**
043 * <p>Build a LiveView fragment</p>
044 *
045 * <p>The packaging rules are as follows :-</p>
046 * <ol>
047 * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
048 *      <ul>
049 *              <li>Archiver-Version: Plexus Archiver</li>
050 *              <li>Built-By: build user</li>
051 *              <li>Build-Jdk: jdk version</li>
052 *              <li>Package-Title: project groupId and name</li>
053 *              <li>Package-Version: project version and buildNumber</li>
054 *              <li>Package-Vendor: project organization name (if set)</li>
055 *              <li>TIBCO-EP-Build-Product-Version: product version</li>
056 *      </ul>
057 * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
058 * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
059 * <li>The project's files are copied into /</li>
060 * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
061 * <li>The project's resources files are copied into /</li>
062 * </ol>
063 *
064 * <p>The plexus archiver is used to create the archive via the maven assembly
065 * plugin.</p>
066 *
067 * <p>Native archives (nar) are included - in this case a mapping is included
068 * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
069 * values.</p>
070 *
071 * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-liveview-fragment.zip</p>
072 */
073@Mojo(name = "package-liveview-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true)
074public class PackageLiveViewFragmentMojo extends BasePackageMojo {
075
076    /**
077     * <p>Eventflow source directories</p>
078     *
079     * <p>If no eventflowDirectories is specified, a single directory of
080     * ${project.basedir}/src/main/eventflow is used.</p>
081     *
082     * <p>Example use in pom.xml:</p>
083     * <img src="uml/eventflowDirectories.svg" alt="pom">
084     *
085     * <p>Example use on commandline:</p>
086     * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
087     *
088     * @since 1.0.0
089     */
090    @Parameter(required = false, property = "eventflowDirectories")
091    File[] eventflowDirectories;
092
093    /**
094     * <p>Liveview source directory</p>
095     *
096     * <p>Example use in pom.xml:</p>
097     * <img src="uml/liveviewDirectory.svg" alt="pom">
098     *
099     * <p>Example use on commandline:</p>
100     * <img src="uml/liveviewDirectory-commandline.svg" alt="pom">
101     *
102     * @since 1.0.0
103     */
104    @Parameter(defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory")
105    File liveviewDirectory;
106
107    @Override
108    public void execute() throws MojoExecutionException {
109        getLog().debug("Creating liveview fragment");
110
111        eventflowDirectories = getOrDefaultSrcMainEventflow(eventflowDirectories);
112
113        prechecks();
114
115        newArchiveGenerator()
116            .withAssemblyStep(assembly -> {
117
118                //  Add files in our source directories.
119                //
120                for (File eventflowDirectory : eventflowDirectories) {
121                    FileSet fileSet = new FileSet();
122                    fileSet.setDirectory(eventflowDirectory.getAbsolutePath());
123                    fileSet.setOutputDirectory("");
124                    assembly.addFileSet(fileSet);
125                }
126
127                FileSet fileSet = new FileSet();
128                fileSet.setDirectory(liveviewDirectory.getAbsolutePath());
129                fileSet.setOutputDirectory("");
130                assembly.addFileSet(fileSet);
131
132            })
133            .generate();
134    }
135}
