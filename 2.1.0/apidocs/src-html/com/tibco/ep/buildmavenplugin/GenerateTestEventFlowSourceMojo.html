031package com.tibco.ep.buildmavenplugin;
032
033import com.tibco.ep.sb.services.build.BuildTarget;
034import org.apache.maven.plugins.annotations.Mojo;
035
036import static org.apache.maven.plugins.annotations.LifecyclePhase.GENERATE_TEST_SOURCES;
037
038
039/**
040 * <p>Typecheck EventFlows and generate java sources</p>
041 *
042 * <p>The plugin will scan the configured EventFlow source and test directories and typecheck/build
043 * them in the dependency order, reporting any typecheck failure.</p>
044 *
045 * <p>It does not build the generated java sources, this is done using a standard maven compiler
046 * plugin execution.</p>
047 *
048 * <p>See {@link GenerateEventFlowSourceMojo} for more information (only differs from the list of
049 * scanned directories).</p>
050 *
051 */
052@Mojo(name = "generate-test-eventflow", defaultPhase = GENERATE_TEST_SOURCES, threadSafe = false)
053public class GenerateTestEventFlowSourceMojo extends BaseGenerateMojo {
054
055    /**
056     * Public constructor
057     */
058    public GenerateTestEventFlowSourceMojo() {
059        super(BuildTarget.TEST);
060    }
061}
