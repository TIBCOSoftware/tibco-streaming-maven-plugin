030package com.tibco.ep.buildmavenplugin;
031
032import java.io.File;
033import java.io.IOException;
034import java.nio.file.Files;
035import java.nio.file.attribute.FileAttribute;
036import java.util.List;
037
038import org.apache.maven.artifact.repository.ArtifactRepository;
039import org.apache.maven.execution.MavenSession;
040import org.apache.maven.plugins.assembly.mojos.AbstractAssemblyMojo;
041import org.apache.maven.project.MavenProject;
042
043/**
044 * Configuration source for assembly
045 *
046 */
047public class ConfigurationSource extends AbstractAssemblyMojo {
048
049    private final MavenProject project;
050    private final ArtifactRepository localRepository;
051    
052    private final File outputDirectory;
053    private final File siteDirectory;
054    private File temporaryDirectory;
055    private File workingDirectory;
056    private final String finalName;
057    private final MavenSession session;
058
059    /**
060     * Constructor
061     * 
062     * @param project project
063     * @param localRepository local repository
064     * @param session session
065     */
066    ConfigurationSource(MavenProject project, ArtifactRepository localRepository, MavenSession session) {
067        this.project = project;
068        this.localRepository = localRepository;
069        this.outputDirectory = new File(project.getBuild().getDirectory());
070        this.siteDirectory = new File(project.getModel().getReporting().getOutputDirectory());
071        
072        File buildDirectory = new File(project.getBuild().getDirectory());
073        if (!buildDirectory.exists()) {
074            if (!buildDirectory.mkdirs()) {
075                getLog().warn("Unable to create directory "+buildDirectory.getAbsolutePath());
076            }
077        }
078        try {
079            this.temporaryDirectory = Files.createTempDirectory(buildDirectory.toPath(), "assemtmp", new FileAttribute[] {}).toFile();
080            this.workingDirectory = Files.createTempDirectory(buildDirectory.toPath(), "assemwork", new FileAttribute[] {}).toFile();
081        } catch (IOException e) {
082            // should never happen
083            e.printStackTrace();
084        }
085        
086        this.finalName = project.getBuild().getFinalName();
087        this.session = session;
088    }
089
090    @Override
091    public File getBasedir() {
092        return project.getBasedir();
093    }
094
095    @Override
096    public String getFinalName() {
097        return finalName;
098    }
099
100    @Override
101    public File getOutputDirectory() {
102        return outputDirectory;
103    }
104
105    @Override
106    public MavenProject getProject() {
107        return project;
108    }
109
110    @Override
111    public List<ArtifactRepository> getRemoteRepositories() {
112        return project.getRemoteArtifactRepositories();
113    }
114
115    @Override
116    public File getSiteDirectory() {
117        return siteDirectory;
118    }
119
120    @Override
121    public File getTemporaryRootDirectory() {
122        return temporaryDirectory;
123    }
124
125    @Override
126    public File getWorkingDirectory() {
127        return workingDirectory;
128    }
129    
130    @Override
131    public ArtifactRepository getLocalRepository() {
132        return localRepository;
133    }
134
135    @Override
136    public MavenSession getMavenSession() {
137        return session;
138    }
139    
140    
141}
