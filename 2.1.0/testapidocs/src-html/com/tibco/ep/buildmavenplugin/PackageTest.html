422    /**
423     * Check if the zip  contains a file
424     *
425     * @param zip      zip archive
426     * @param filename filename
427     * @return true if it does contain it
428     */
429    boolean zipContains(String zip, String filename) {
430        boolean found = false;
431
432        ZipFile zipFile;
433        try {
434            zipFile = new ZipFile(zip);
435
436            Enumeration<? extends ZipEntry> zipEntries = zipFile.entries();
437            String fname;
438            while (zipEntries.hasMoreElements()) {
439                fname = ((ZipEntry) zipEntries.nextElement()).getName();
440                if (fname.equals(filename)) {
441                    found = true;
442                }
443            }
444            zipFile.close();
445        } catch (IOException e) {
446            // TODO Auto-generated catch block
447            e.printStackTrace();
448        }
449
450        return found;
451    }

453    /**
454     * Check if the file contains test
455     *
456     * @param filename gile
457     * @param text     text to look for
458     * @return true if it does contain it
459     */
460    boolean fileContains(String filename, String text) {
461        boolean found = false;
462        try {
463            File file = new File(filename);
464            Scanner scanner = new Scanner(file);
465            while (scanner.hasNextLine()) {
466                String line = scanner.nextLine();
467                if (line.contains(text)) {
468                    found = true;
469                }
470            }
471            scanner.close();
472        } catch (FileNotFoundException e) {
473        }
474        return found;
475    }
