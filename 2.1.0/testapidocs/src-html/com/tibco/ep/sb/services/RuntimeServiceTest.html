077    /**
078     * Tests for build service stubs
079     */
080    @Test
081    public void testStubbedBuildServiceImplementation() {
082
083        IRuntimeBuildService build = RuntimeServices.getBuildService(getClass().getClassLoader());
084        assertThat(build).isNotNull();
085
086        BuildParameters parameters = new BuildParameters();
087
088        Map<String, String> properties = new HashMap<>();
089        properties.put("MyProp", "MyVal");
090        parameters
091            .withBuildDirectory(Paths.get("buildDir"))
092            .withProjectCompileClassPath(Arrays.asList(
093                Paths.get("first.jar"),
094                Paths.get("path", "to", "classes")))
095            .withProjectTestCompileClassPath(Collections.singletonList(Paths.get("test.jar")))
096            .withCompilerProperties(properties)
097            .withConfigurationDirectory(Paths.get("src", "main", "configurations"))
098            .withTestConfigurationDirectory(Paths.get("src", "test", "configurations"))
099            .withSourcePaths(Collections.singletonList(Paths.get("src", "main")))
100            .withTestSourcePaths(Collections.singletonList(Paths.get("src", "test")));
101
102        BuildNotifier handler = new BuildNotifier();
103        build.build("MyBuild", BuildTarget.MAIN, parameters, handler);
104
105        assertThat(handler.getStartedEntity()).isNotEmpty();
106        assertThat(handler.getResult().getElapsedTimeMillis()).isGreaterThan(0);
107        assertThat(handler.getResult().getEntityName()).isNotEmpty();
108        assertThat(handler.getResult().getEntityPath()).isNotNull();
109        assertThat(handler.getResult().getException()).isNotNull();
110    }

112    /**
113     * Tests for admin service stubs
114     */
115    @Test
116    public void testStubbedAdminServiceImplementation() {
117
118        IRuntimeAdminService admin = RuntimeServices.getAdminService(getClass().getClassLoader());
119        assertThat(admin).isNotNull();
120
121        //  Context.
122        //
123        IContext ctx = admin.newContext(Paths.get("/some/path"));
124        ctx.withTracingEnabled();
125        ctx.clearEnvironment();
126
127        Map<String, String> environment = new HashMap<>();
128        environment.put("key", "value");
129        ctx.withEnvironment(environment);
130
131        //  Node.
132        //
133        AbstractNodeBuilder nodeBuilder = ctx.newNode("MyNode")
134            .withHostName("MyNodeHost")
135            .withAdministrationPort(123)
136            .withUserName("user")
137            .withPassword("password")
138            .withAdditionalDiscoveryHost("discovery_host");
139        assertThat(nodeBuilder.getHostName()).isEqualTo("MyNodeHost");
140        assertThat(nodeBuilder.getAdministrationPort()).isEqualTo(123);
141        INode node = nodeBuilder.build();
142
143        //  Destination.
144        //
145        AbstractDestinationBuilder destinationBuilder = ctx.newDestination()
146            .withName("destination_name")
147            .withUserName("username")
148            .withPassword("password")
149            .withAdditionalDiscoveryHost("discovery_host1")
150            .withAdditionalDiscoveryHost("discovery_host2");
151        assertThat(destinationBuilder.getName()).isEqualTo("destination_name");
152        assertThat(destinationBuilder.getUserName()).isEqualTo("username");
153        assertThat(destinationBuilder.getPassword()).isEqualTo("password");
154        assertThat(destinationBuilder.getDiscoveryHosts())
155            .contains("discovery_host1", "discovery_host2");
156        assertThat(destinationBuilder.getContext()).isEqualTo(ctx);
157
158        IDestination destination = destinationBuilder.build();
159        destination.setDiscoveryPort(1234);
160        AbstractCommandBuilder commandBuilder = destination.newCommand("command", "target")
161            .withCommand("command")
162            .withTarget("target")
163            .withDestination(destination);
164        assertThat(commandBuilder.getCommand()).isEqualTo("command");
165        assertThat(commandBuilder.getTarget()).isEqualTo("target");
166        assertThat(commandBuilder.getDestination()).isEqualTo(destination);
167
168        //  Command.
169        //
170        ICommand commandWithDestination = commandBuilder.build();
171        assertThat(commandWithDestination.getDestination()).isEqualTo(destination);
172        DummyNotifier notifier = new DummyNotifier();
173        commandWithDestination.execute(environment, notifier);
174        assertThat(commandWithDestination.waitForCompletion()).isEqualTo(0);
175        assertThat(notifier.started).isTrue();
176        assertThat(notifier.complete).isTrue();
177        assertThat(notifier.info).isNotEmpty();
178        commandWithDestination.cancel();
179
180        //  Deploy command.
181        //
182        AbstractDeployFragmentCommandBuilder deployCommandBuilder = destination
183            .newDeployFragmentCommand(FragmentType.EVENTFLOW, "target")
184            .withExecutionOptions(Arrays.asList("exec1", "exec2"))
185            .withApplicationArguments(Arrays.asList("app1", "app2"));
186        assertThat(deployCommandBuilder.getApplicationArguments()).containsExactly("app1", "app2");
187        assertThat(deployCommandBuilder.getExecutionOptions()).containsExactly("exec1", "exec2");
188        assertThat(deployCommandBuilder.getFragmentType()).isEqualTo(FragmentType.EVENTFLOW);
189        IDeployFragmentCommand deployFragmentCommand = deployCommandBuilder.build();
190        assertThat(deployFragmentCommand).isNotNull();
191
192        //  Browse services command
193        //
194        IBrowseServicesCommand browseServicesCommand = ctx
195            .newBrowseServicesCommand().build();
196        assertThat(browseServicesCommand).isNotNull();
197
198        //  Install node command
199        //
200        IInstallNodeCommand installNodeCommand = node
201            .newInstallNodeCommand()
202            .build();
203        assertThat(installNodeCommand).isNotNull();
204    }

251    private class BuildNotifier implements IBuildNotifier {
252        private String startedEntity;
253        private BuildResult result;
254
255        public String getStartedEntity() {
256            return startedEntity;
257        }
258
259        public BuildResult getResult() {
260            return result;
261        }
262
263        @Override
264        public void onStarted(String entityName, String entityExtension) {
265            this.startedEntity = entityName;
266        }
267
268        @Override
269        public void onCompleted(BuildResult result) {
270            this.result = result;
271        }
272    }
