8   package com.tibco.ep.buildmavenplugin;
9   
10  import org.apache.maven.plugin.MojoExecutionException;
11  import org.apache.maven.plugins.annotations.Mojo;
12  import org.apache.maven.plugins.annotations.ResolutionScope;
13  
14  import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
15  
16  /**
17   * <p>Build a TCS fragment</p>
18   *
19   * <p>The packaging rules are as follows :-</p>
20   * <ol>
21   * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
22   *      <ul>
23   *              <li>Archiver-Version: Plexus Archiver</li>
24   *              <li>Built-By: build user</li>
25   *              <li>Build-Jdk: jdk version</li>
26   *              <li>Package-Title: project groupId and name</li>
27   *              <li>Package-Version: project version and buildNumber</li>
28   *              <li>Package-Vendor: project organization name (if set)</li>
29   *              <li>TIBCO-EP-Build-Product-Version: product version</li>
30   *      </ul>
31   * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
32   * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
33   * <li>The project's files are copied into /</li>
34   * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
35   * <li>The project's resources files are copied into /</li>
36   * </ol>
37   *
38   * <p>The plexus archiver is used to create the archive via the maven assembly
39   * plugin.</p>
40   *
41   * <p>Native archives (nar) are included - in this case a mapping is included
42   * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
43   * values.</p>
44   *
45   * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-tcs-fragment.zip</p>
46   */
47  @Mojo(name = "package-tcs-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true)
48  public class PackageTCSFragmentMojo extends BasePackageMojo {
49  
50      @Override
51      public void execute() throws MojoExecutionException {
52          getLog().debug("Creating TCS fragment");
53  
54          prechecks();
55  
56          newArchiveGenerator().generate();
57      }
58  }
