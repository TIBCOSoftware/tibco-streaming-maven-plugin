008package com.tibco.ep.buildmavenplugin;
009
010import org.apache.maven.plugin.MojoExecutionException;
011import org.apache.maven.plugins.annotations.Mojo;
012import org.apache.maven.plugins.annotations.ResolutionScope;
013
014import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
015
016/**
017 * <p>Build a TCS fragment</p>
018 *
019 * <p>The packaging rules are as follows :-</p>
020 * <ol>
021 * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
022 *      <ul>
023 *              <li>Archiver-Version: Plexus Archiver</li>
024 *              <li>Built-By: build user</li>
025 *              <li>Build-Jdk: jdk version</li>
026 *              <li>Package-Title: project groupId and name</li>
027 *              <li>Package-Version: project version and buildNumber</li>
028 *              <li>Package-Vendor: project organization name (if set)</li>
029 *              <li>TIBCO-EP-Build-Product-Version: product version</li>
030 *      </ul>
031 * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
032 * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
033 * <li>The project's files are copied into /</li>
034 * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
035 * <li>The project's resources files are copied into /</li>
036 * </ol>
037 *
038 * <p>The plexus archiver is used to create the archive via the maven assembly
039 * plugin.</p>
040 *
041 * <p>Native archives (nar) are included - in this case a mapping is included
042 * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
043 * values.</p>
044 *
045 * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-tcs-fragment.zip</p>
046 */
047@Mojo(name = "package-tcs-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true)
048public class PackageTCSFragmentMojo extends BasePackageMojo {
049
050    @Override
051    public void execute() throws MojoExecutionException {
052        getLog().debug("Creating TCS fragment");
053
054        prechecks();
055
056        newArchiveGenerator().generate();
057    }
058}
