030package com.tibco.ep.buildmavenplugin.surefire;
031
032import java.util.Map;
033import org.apache.maven.surefire.report.ReporterException;
034import org.apache.maven.surefire.report.ReporterManager;
035import org.apache.maven.surefire.testset.SurefireTestSet;
036import org.apache.maven.surefire.testset.TestSetFailedException;
037
038import java.lang.reflect.Modifier;
039import java.util.Arrays;
040import java.util.Collections;
041import java.util.HashMap;
042import java.util.ResourceBundle;
043import org.apache.maven.surefire.Surefire;
044import org.apache.maven.surefire.report.ReportEntry;
045import org.apache.maven.surefire.suite.SurefireTestSuite;
046
047/**
048 * Test suite to run an explicit set of named unit test classes
049 */
050public class TestSuite implements SurefireTestSuite {
051    /**
052     * SureFire resource bundle
053     */
054    protected final static ResourceBundle bundle = ResourceBundle.getBundle(Surefire.SUREFIRE_BUNDLE_NAME);
055
056    /**
057     * Array of test class names (in Java binary format)
058     */
059    private final String[] m_testClasses;
060
061    /**
062     * The set of known tests sets, indexed by test set name.
063     */
064    private HashMap<String, SurefireTestSet> m_testMap;
065
066    /**
067     * Create a test suite runner for a series of test classes.
068     * 
069     * @param testClasses
070     *            array of test class names
071     */
072    public TestSuite(String[] testClasses) {
073        m_testClasses = Arrays.copyOf(testClasses, testClasses.length);
074    }
075
076    /**
077     * Execute test suite
078     * 
079     * @param reporterManager
080     *            Report manager
081     * @param classLoader
082     *            class loader
083     * @throws ReporterException Reporting exception
084     * @throws TestSetFailedException Test failure exception
085     */
086    @Override
087    public void execute(ReporterManager reporterManager, ClassLoader classLoader)
088            throws ReporterException, TestSetFailedException {
089        for (SurefireTestSet testSet : m_testMap.values()) {
090            executeTestSet(testSet, reporterManager, classLoader);
091            reporterManager.reset();
092        }
093    }
094
095    /**
096     * Execute test test
097     * 
098     * @param testSetName
099     *            Name of test set
100     * @param reporterManager
101     *            Report manager
102     * @param classLoader
103     *            Class loader
104     * @throws ReporterException Reporting exception
105     * @throws TestSetFailedException Test failure exception
106     */
107    @Override
108    public void execute(String testSetName, ReporterManager reporterManager, ClassLoader classLoader)
109            throws ReporterException, TestSetFailedException {
110        SurefireTestSet testSet = m_testMap.get(testSetName);
111        if (testSet == null) {
112            // FIX THIS: is this even possible?
113            throw new TestSetFailedException("Test set '" + testSetName + "' does not exist.");
114        }
115        executeTestSet(testSet, reporterManager, classLoader);
116    }
117
118    /**
119     * Number of test cases
120     * 
121     * @return Number of test cases
122     */
123    @Override
124    public int getNumTests() {
125        return m_testClasses.length;
126    }
127
128    /**
129     * Locate test sets
130     * 
131     * @param classLoader
132     *            Class loader
133     * @return Map of test sets
134     * @throws TestSetFailedException Test failure exception
135     */
136    @Override
137    public Map<String, SurefireTestSet> locateTestSets(ClassLoader classLoader)
138        throws TestSetFailedException {
139        m_testMap = new HashMap<>();
140
141        for (String className : m_testClasses) {
142            Class<?> testClass;
143            try {
144                testClass = classLoader.loadClass(className);
145            } catch (ClassNotFoundException e) {
146                throw new TestSetFailedException("Unable to load test class '" + className + "'", e);
147            }
148
149            if (!Modifier.isAbstract(testClass.getModifiers())) {
150                m_testMap.put(className, new TestSet(testClass));
151            }
152        }
153        return Collections.unmodifiableMap(m_testMap);
154    }
155
156    private void executeTestSet(SurefireTestSet testSet, ReporterManager reporterManager, ClassLoader classLoader)
157            throws ReporterException, TestSetFailedException {
158        reporterManager.testSetStarting(
159                new ReportEntry(this.getClass().getName(), testSet.getName(), bundle.getString("testSetStarting")));
160
161        testSet.execute(reporterManager, classLoader);
162
163        reporterManager.testSetCompleted(new ReportEntry(this.getClass().getName(), testSet.getName(),
164                bundle.getString("testSetCompletedNormally")));
165
166        reporterManager.reset();
167    }
168}
