058/**
059 * <p>Install product zip and tgz artifacts if not already installed.</p>
060 * 
061 * <p>This goal scans through the project dependencies and if a product zip
062 * or tgz artifact is not installed, it is installed into the $TIBCO_EP_HOME
063 * directory.</p>
064 * 
065 * <p>Dependencies with scope <b>provided</b> are skipped.</p>
066 * 
067 * <p>A <b>markersDirectory</b> is used to track if the artifact has already
068 * been installed - hence manually removing this directory will cause the
069 * artifacts to be re-installed.  This directory is only created if there is
070 * a need to install an artifact.</p>
071 *
072 * <p>To detect a manual installation of the product, a file specified by
073 * <b>productValidationFile</b> is used.  If this file exists, then the
074 * product installation is not attempted.</p>
075 * 
076 * <p>If the plugin does install a zip, then an md5 sum of the original zip
077 * is saved - this allows installation of a newer version of the zip at a
078 * later time.  This can happen with SNAPSHOTS.</p>
079 */
080@Mojo(name = "install-product", defaultPhase = VALIDATE, threadSafe = true)
081public class InstallProductMojo extends BaseMojo {
082
083    /**
084     * <p>Directory to store flag files.</p>
085     * 
086     * <p>Relative to productHome.<p>
087     * 
088     * <p>Example use in pom.xml:</p>
089     * <img src="uml/installproduct-markersDirectory.svg" alt="pom">
090     * 
091     * @since 1.0.0
092     */
093    @Parameter(defaultValue="dependency-maven-plugin-markers")
094    String markersDirectory;
095
096    /**
097     * <p>File to check if DTM product has been installed externally - if this exists,
098     * then product installation is not attempted.</p>
099     * 
100     * <p>Relative to productHome.<p>
101     * 
102     * <p>Example use in pom.xml:</p>
103     * <img src="uml/installproduct-dtmProductValidationFile.svg" alt="pom">
104     * 
105     * @since 1.0.0
106     */
107    @Parameter(defaultValue="distrib/tibco/dtm/deploy/dtm.kds")
108    String dtmProductValidationFile;
109
110    /**
111     * <p>File to check if SB product has been installed externally - if this exists,
112     * then product installation is not attempted.</p>
113     * 
114     * <p>Relative to productHome.<p>
115     *
116     * <p>Example use in pom.xml:</p>
117     * <img src="uml/installproduct-sbProductValidationFile.svg" alt="pom">
118     * 
119     * @since 1.0.0
120     */
121    @Parameter(defaultValue="distrib/tibco/sb/deploy/sb.kds")
122    String sbProductValidationFile;
123
124    /**
125     * <p>File to check if DTM support has been installed externally - if this exists,
126     * then support installation is not attempted.</p>
127     * 
128     * <p>Relative to productHome.<p>
129     *
130     * <p>Example use in pom.xml:</p>
131     * <img src="uml/installproduct-dtmSupportValidationFile.svg" alt="pom">
132     * 
133     * @since 1.2.0
134     */
135    @Parameter(defaultValue="distrib/tibco/devbin/epadmin")
136    String dtmSupportValidationFile;
137
138    /**
139     * <p>Set this to 'true' to skip running tests, but still compile them.</p>
140     * 
141     * <p>Example use in pom.xml:</p>
142     * <img src="uml/skipTests.svg" alt="pom">
143     * 
144     * <p>Example use on commandline:</p>
145     * <img src="uml/skipTests-commandline.svg" alt="pom">
146     * 
147     * @since 1.0.0
148     */
149    @Parameter(property = "skipTests", defaultValue = "false")
150    boolean skipTests;

152    public void execute() throws MojoExecutionException {
153        getLog().debug("Install product");
154
155        prechecks();
156
157        for (Artifact artifact : getProjectDependencies("zip")) {
158            
159            final String artifactAsString = artifact.toString();
160            
161            // skip provided
162            if (artifact.getScope().equals(Artifact.SCOPE_PROVIDED)) {
163                getLog().debug("Ignoring provided artifact " + artifactAsString);
164                continue;
165            }
166            
167            // Platform and main attachments only.
168            if (!isPlatformArtifact(artifact)) {
169                getLog().debug("Ignoring non-platform artifact " + artifactAsString);
170                continue;
171            }
172            
173            if (null != artifact.getClassifier()) {
174                getLog().debug("Ignore classified artifact " + artifactAsString);
175                continue;
176            }
177
178            getLog().debug(artifactAsString+" local path = "+getArtifactPath(artifact));
179
180            File sourceFile = new File(getArtifactPath(artifact));
181            
182            boolean forceReplace = false;
183
184            File markersFile = new File(productHome, markersDirectory+File.separator+getArtifactName(artifact)+".marker");
185
186            if (isInstalled(markersFile, sourceFile, artifact)) {
187                continue;
188            }
189            
190            // now we need check for directories existing, write access and create markers directory
191            
192            productHome.mkdirs();
193            if (!productHome.exists()) {
194                throw new MojoExecutionException("The product directory "+productHome+" does not exists or cannot be created");
195            }
196            if (! productHome.canWrite()) {
197                throw new MojoExecutionException("The product directory "+productHome+" cannot be written to for markers directory");
198            }
199            
200
201            Locker locker = new Locker(productHome);
202            try {
203                
204
205                File markersDir = new File(productHome, markersDirectory);
206
207                // Create markers directory of it doesn't exist
208                //
209                if (!markersDir.exists() && !markersDir.mkdirs()) {
210                    throw new MojoExecutionException("Unable to create markers directory "+markersDir.getAbsolutePath());
211                }
212                
213                // we might have held the lock for a long time, so check again before doing the big work
214                if (isInstalled(markersFile, sourceFile, artifact)) {
215                    locker.release();
216                    continue;
217                }
218                
219                getLog().info("Installing "+artifactAsString+" to "+productHome);
220
221                if (!productHome.exists() && !productHome.mkdirs()) {
222                    locker.release();
223                    throw new MojoExecutionException("Unable to create product directory "+productHome.getAbsolutePath());
224                }
225
226                // delete any old files - this avoid permission issues when re-extracting
227                //
228                ZipInputStream zis = null;
229                try {
230                    zis = new ZipInputStream(new FileInputStream(sourceFile));
231                    ZipEntry zipEntry = zis.getNextEntry();
232                    while(zipEntry != null){
233                        File oldFile = new File(productHome, zipEntry.getName());
234                        if (oldFile.exists() && oldFile.isFile()) {
235                            getLog().debug("File "+oldFile+" already exists so deleting");
236                            if (!oldFile.delete()) {
237                                getLog().warn("File "+oldFile+" failed to be deleted");
238                            }
239                        }
240                        zipEntry = zis.getNextEntry();
241                    }
242                    zis.closeEntry();
243                } catch (IOException e) {
244                } finally {
245                    if (zis != null) {
246                        try {
247                            zis.close();
248                        } catch (IOException e) {
249                        }
250                    }
251                }
252
253                // and extract
254                //
255                ZipUnArchiver ua = new ZipUnArchiver();
256                ua.setSourceFile(sourceFile);
257                ua.setDestDirectory(productHome);
258                ua.enableLogging(new ConsoleLogger(Logger.LEVEL_INFO, "install-product"));
259                if (forceReplace) {
260                    ua.setOverwrite(true);
261                }
262                ua.extract();
263
264                // we installed this, so create a md5 of original zip
265                //
266                String md5 = this.md5(sourceFile);
267                BufferedWriter output = null;
268                try {
269                    output = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(markersFile), StandardCharsets.UTF_8));
270                    output.write(md5);
271                } catch (IOException e) {
272                    getLog().warn("Unable to save zip checksum - "+e.getMessage());
273                } finally {
274                    if (output != null) {
275                        try {
276                            output.close();
277                        } catch (IOException e) {
278                        }
279                    }
280                }
281
282            } finally {
283                locker.release();
284            }
285
286        }
287    }
288    
289    // return true if already installed, false otherwise
290    //
291    private boolean isInstalled(File markersFile, File sourceFile, Artifact artifact) {
292        boolean forceReplace = false;
293        
294        if (markersFile.exists()) {
295            // if we have an md5 of the installation and its not the same as the zip we are installing, 
296            // install anyway
297            //
298            // this can happen with SNAPSHOT builds
299            //
300            if (markersFile.length() > 0) {
301                BufferedReader input = null;
302                try {
303                    input = new BufferedReader(new InputStreamReader(new FileInputStream(markersFile), StandardCharsets.UTF_8));
304                    String oldmd5 = input.readLine();
305                    if (oldmd5 == null) {
306                        getLog().warn("Unable to verify zip checksum");
307                    } else {
308                        String md5 = this.md5(sourceFile);
309                        if (!oldmd5.equals(md5)) {
310                            getLog().warn("Previous installation at "+productHome+" is old - overwriting");
311                            forceReplace = true;
312                        }
313                    }
314                } catch (IOException e) {
315                    getLog().warn("Unable to verify zip checksum - "+e.getMessage());
316                } finally {
317                    if (input != null) {
318                        try {
319                            input.close();
320                        } catch (IOException e) {
321                        }
322                    }
323                }
324            }
325
326            if (!forceReplace) {
327                getLog().info(artifact.toString()+" already installed by maven to "+productHome);
328                return true;
329            }
330        }
331
332        if (!forceReplace && isSBProduct(artifact) && new File(productHome, sbProductValidationFile).exists()) {
333            getLog().info(artifact.toString()+" already installed manually to "+productHome);
334            return true;
335        }
336        if (!forceReplace && isDTMProduct(artifact) && new File(productHome, dtmProductValidationFile).exists()) {
337            getLog().info(artifact.toString()+" already installed manually to "+productHome);
338            return true;
339        }
340        if (!forceReplace && isDTMSupport(artifact) && new File(productHome, dtmSupportValidationFile).exists()) {
341            getLog().info(artifact.toString()+" already installed manually to "+productHome);
342            return true;
343        }
344
345        if (!sourceFile.exists()) {
346            getLog().debug(getArtifactPath(artifact)+" not downloaded yet ... skipping unpack");
347            return true;
348        }
349        
350        return false;
351    }
352    
353    
354    private String md5(final File sourceFile) {
355        
356        String md5 = "";
357        InputStream input = null;
358        try {
359            MessageDigest md5Local = MessageDigest.getInstance("MD5");
360            input = new FileInputStream(sourceFile.toString());
361            byte[] buffer = new byte[4096];
362            int n;
363            while ((n = input.read(buffer)) != -1) {
364                md5Local.update(buffer, 0, n);
365            }
366            byte[] digest = md5Local.digest();
367            BigInteger bi = new BigInteger(1, digest);
368            md5 = String.format("%0" + (digest.length << 1) + "X", bi);
369        } catch (NoSuchAlgorithmException e) {
370            getLog().warn("Unable to save zip checksum - "+e.getMessage());
371        } catch (IOException e) {
372            getLog().warn("Unable to save zip checksum - "+e.getMessage());
373        } finally {
374            if (input != null) {
375                try {
376                    input.close();
377                } catch (IOException e) {
378                }
379            }
380        }
381        
382        return md5;
383    }
