028package com.tibco.ep.buildmavenplugin.surefire;
029
030/**
031 * Message identifiers for messages sent by the RemoteTestRunner.
032 *
033 * see org.eclipse.jdt.internal.junit.runner.RemoteTestRunner
034 */
035public class JUnitViewMessageIds {
036    /**
037     * The header length of a message, all messages have a fixed header length
038     */
039    public static final int MSG_HEADER_LENGTH = 8;
040
041    /**
042     * Notification that a test trace has started. The end of the trace is signaled
043     * by a TRACE_END message. In between the TRACE_START and TRACE_END the stack
044     * trace is submitted as multiple lines.
045     */
046    public static final String TRACE_START = "%TRACES "; //$NON-NLS-1$
047    /**
048     * Notification that a trace ends.
049     */
050    public static final String TRACE_END = "%TRACEE "; //$NON-NLS-1$
051    /**
052     * Notification that the expected result has started. The end of the expected
053     * result is signaled by a Trace_END.
054     */
055    public static final String EXPECTED_START = "%EXPECTS"; //$NON-NLS-1$
056    /**
057     * Notification that an expected result ends.
058     */
059    public static final String EXPECTED_END = "%EXPECTE"; //$NON-NLS-1$
060    /**
061     * Notification that the expected result has started. The end of the expected
062     * result is signaled by a Trace_END.
063     */
064    public static final String ACTUAL_START = "%ACTUALS"; //$NON-NLS-1$
065    /**
066     * Notification that an expected result ends.
067     */
068    public static final String ACTUAL_END = "%ACTUALE"; //$NON-NLS-1$
069    /**
070     * Notification that a trace for a reran test has started. The end of the trace
071     * is signaled by a RTrace_END message.
072     */
073    public static final String RTRACE_START = "%RTRACES"; //$NON-NLS-1$
074    /**
075     * Notification that a trace of a reran trace ends.
076     */
077    public static final String RTRACE_END = "%RTRACEE"; //$NON-NLS-1$
078    /**
079     * Notification that a test run has started. MessageIds.TEST_RUN_START +
080     * testCount.toString + " " + version
081     */
082    public static final String TEST_RUN_START = "%TESTC  "; //$NON-NLS-1$
083    /**
084     * Notification that a test has started. MessageIds.TEST_START + testID + "," +
085     * testName
086     */
087    public static final String TEST_START = "%TESTS  "; //$NON-NLS-1$
088    /**
089     * Notification that a test has started. TEST_END + testID + "," + testName
090     */
091    public static final String TEST_END = "%TESTE  "; //$NON-NLS-1$
092    /**
093     * Notification that a test had a error. TEST_ERROR + testID + "," + testName.
094     * After the notification follows the stack trace.
095     */
096    public static final String TEST_ERROR = "%ERROR  "; //$NON-NLS-1$
097    /**
098     * Notification that a test had a failure. TEST_FAILED + testID + "," +
099     * testName. After the notification follows the stack trace.
100     */
101    public static final String TEST_FAILED = "%FAILED "; //$NON-NLS-1$
102    /**
103     * Notification that a test run has ended. TEST_RUN_END +
104     * elapsedTime.toString().
105     */
106    public static final String TEST_RUN_END = "%RUNTIME"; //$NON-NLS-1$
107    /**
108     * Notification that a test run was successfully stopped.
109     */
110    public static final String TEST_STOPPED = "%TSTSTP "; //$NON-NLS-1$
111    /**
112     * Notification that a test was reran. TEST_RERAN + testId + " " + testClass + "
113     * " + testName + STATUS. Status = "OK" or "FAILURE".
114     */
115    public static final String TEST_RERAN = "%TSTRERN"; //$NON-NLS-1$
116    /**
117     * Notification about a test inside the test suite. TEST_TREE + testId + "," +
118     * testName + "," + isSuite + "," + testcount isSuite = "true" or "false"
119     */
120    public static final String TEST_TREE = "%TSTTREE"; //$NON-NLS-1$
121    /**
122     * Request to stop the current test run.
123     */
124    public static final String TEST_STOP = ">STOP   "; //$NON-NLS-1$
125    /**
126     * Request to rerun a test. TEST_RERUN + testId + " " + testClass + " "+testName
127     */
128    public static final String TEST_RERUN = ">RERUN  "; //$NON-NLS-1$
129
130    /**
131     * MessageFormat to encode test method identifiers: testMethod(testClass)
132     */
133    public static final String TEST_IDENTIFIER_MESSAGE_FORMAT = "{0}({1})"; //$NON-NLS-1$
134
135    /**
136     * Test identifier prefix for ignored tests.
137     */
138    public static final String IGNORED_TEST_PREFIX = "@Ignore: "; //$NON-NLS-1$
139
140    /**
141     * Test identifier prefix for tests with assumption failures.
142     */
143    public static final String ASSUMPTION_FAILED_TEST_PREFIX = "@AssumptionFailure: "; //$NON-NLS-1$
144}
