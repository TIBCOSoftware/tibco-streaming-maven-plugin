030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.logging.Log;
033import org.slf4j.event.Level;
034
035/**
036 * Simulated maven logger to catch the actual output
037 */
038public class SimulatedLog implements Log {
039    private final boolean verbose;
040    private StringBuilder debugText;
041    private StringBuilder infoText;
042    private StringBuilder errorText;
043    private StringBuilder warnText;
044
045    /**
046     * Constructor
047     *
048     * @param verbose true to log all output
049     */
050    SimulatedLog(boolean verbose) {
051        this.verbose = verbose;
052        reset();
053    }
054
055    /**
056     * Reset stored stings
057     */
058    public void reset() {
059        debugText = new StringBuilder();
060        infoText = new StringBuilder();
061        errorText = new StringBuilder();
062        warnText = new StringBuilder();
063    }
064
065    /**
066     * Get stored debug log
067     *
068     * @return debug log
069     */
070    public String getDebugLog() {
071        return debugText.toString();
072    }
073
074    /**
075     * Get stored info log
076     *
077     * @return info log
078     */
079    public String getInfoLog() {
080        return infoText.toString();
081    }
082
083    /**
084     * Get stored error log
085     *
086     * @return error log
087     */
088    public String getErrorLog() {
089        return errorText.toString();
090    }
091
092    /**
093     * Get stored warn log
094     *
095     * @return warn log
096     */
097    public String getWarnLog() {
098        return warnText.toString();
099    }
100
101    private void log(Level level, CharSequence msg, Throwable error) {
102
103        if (verbose) {
104            //  DO NOT use SLF4J here, this will cause recursion because of the SLF4J Maven binding.
105            //
106            System.out.println("Verbose SimulatedLog: " + level + ": " + msg
107                + (error != null ? " error = " + error : ""));
108        }
109
110        StringBuilder builder = null;
111        switch (level) {
112
113            case ERROR:
114                builder = errorText;
115                break;
116            case WARN:
117                builder = warnText;
118                break;
119            case INFO:
120                builder = infoText;
121                break;
122            case DEBUG:
123                builder = debugText;
124                break;
125            case TRACE:
126            default:
127                throw new IllegalArgumentException("Bad level: " + level);
128        }
129
130        if (msg != null) {
131            builder.append(msg).append("\n");
132        }
133        if (error != null) {
134            builder.append(error).append("\n");
135        }
136
137    }
138
139    @Override
140    public void debug(CharSequence content) {
141        log(Level.DEBUG, content, null);
142    }
143
144    @Override
145    public void debug(Throwable error) {
146        log(Level.DEBUG, null, error);
147    }
148
149    @Override
150    public void debug(CharSequence content, Throwable error) {
151        log(Level.DEBUG, content, error);
152    }
153
154    @Override
155    public void info(CharSequence content) {
156        log(Level.INFO, content, null);
157    }
158
159    @Override
160    public void info(Throwable error) {
161        log(Level.INFO, null, error);
162    }
163
164    @Override
165    public void info(CharSequence content, Throwable error) {
166        log(Level.INFO, content, error);
167    }
168
169    @Override
170    public void warn(CharSequence content) {
171        log(Level.WARN, content, null);
172    }
173
174    @Override
175    public void warn(Throwable error) {
176        log(Level.WARN, null, error);
177    }
178
179    @Override
180    public void warn(CharSequence content, Throwable error) {
181        log(Level.WARN, content, error);
182    }
183
184    @Override
185    public void error(CharSequence content) {
186        log(Level.ERROR, content, null);
187    }
188
189    @Override
190    public void error(Throwable error) {
191        log(Level.ERROR, null, error);
192    }
193
194    @Override
195    public void error(CharSequence content, Throwable error) {
196        log(Level.ERROR, content, error);
197    }
198
199    @Override
200    public boolean isDebugEnabled() {
201        return true;
202    }
203
204    @Override
205    public boolean isErrorEnabled() {
206        return true;
207    }
208
209    @Override
210    public boolean isInfoEnabled() {
211        return true;
212    }
213
214    @Override
215    public boolean isWarnEnabled() {
216        return true;
217    }
218}
