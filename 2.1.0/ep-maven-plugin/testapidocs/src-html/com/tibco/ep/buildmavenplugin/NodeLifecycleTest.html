030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.testing.MojoRule;
033import org.apache.maven.plugin.testing.resources.TestResources;
034import org.junit.After;
035import org.junit.Assert;
036import org.junit.Rule;
037import org.junit.Test;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041import java.io.File;
042
043/**
044 * Node lifecycle - start and stop
045 */
046public class NodeLifecycleTest extends BetterAbstractMojoTestCase {
047
048    private static final Logger LOGGER = LoggerFactory.getLogger(NodeLifecycleTest.class);
049    /**
050     * rule
051     */
052    @Rule
053    public MojoRule rule = new MojoRule();
054    /**
055     * resources
056     */
057    @Rule
058    public TestResources resources = new TestResources();
059    SimulatedLog simulatedLog = new SimulatedLog(false);
060
061    /**
062     * Clean up
063     */
064    @After
065    public void postTest() throws Exception {
066
067        File stop = new File("target/projects/lifecycle", "stop.xml");
068        Assert.assertNotNull(stop);
069        Assert.assertTrue(stop.exists());
070
071        LOGGER.info("   Stop nodes");
072        StopNodesMojo stopNodes = (StopNodesMojo) lookupConfiguredMojo(stop, "stop-nodes");
073        Assert.assertNotNull(stopNodes);
074        stopNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
075        simulatedLog.reset();
076        stopNodes.setLog(simulatedLog);
077        stopNodes.execute();
078        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
079        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
080        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Node removed"));
081    }
082
083    /**
084     * Install product - do this first
085     *
086     * @throws Exception on error
087     */
088    @Test
089    public void testNodeLifecycle() throws Exception {
090        LOGGER.info("Install Product");
091
092        File pom = new File("target/projects", "pom.xml");
093        Assert.assertNotNull(pom);
094        Assert.assertTrue(pom.exists());
095
096        InstallProductMojo installProduct = (InstallProductMojo) lookupConfiguredMojo(pom, "install-product");
097        Assert.assertNotNull(installProduct);
098        installProduct.setLog(simulatedLog);
099        installProduct.execute();
100        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
101        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
102
103        LOGGER.info("Test lifecycle");
104        simulatedLog = new SimulatedLog(false);
105
106        pom = new File("target/projects/lifecycle", "pom.xml");
107        Assert.assertNotNull(pom);
108        Assert.assertTrue(pom.exists());
109
110        File status = new File("target/projects/lifecycle", "status.xml");
111        Assert.assertNotNull(status);
112        Assert.assertTrue(status.exists());
113
114        File status2 = new File("target/projects/lifecycle", "status2.xml");
115        Assert.assertNotNull(status);
116        Assert.assertTrue(status.exists());
117
118        File status3 = new File("target/projects/lifecycle", "status3.xml");
119        Assert.assertNotNull(status);
120        Assert.assertTrue(status.exists());
121
122        LOGGER.info("   Start nodes");
123        StartNodesMojo startNodes = (StartNodesMojo) lookupConfiguredMojo(pom, "start-nodes");
124        Assert.assertNotNull(startNodes);
125        startNodes.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
126        simulatedLog.reset();
127        startNodes.setLog(simulatedLog);
128        startNodes.execute();
129        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
130        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
131        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Node started"));
132
133        LOGGER.info("   Display nodes");
134        AdministrationMojo administrationMojo = (AdministrationMojo) lookupConfiguredMojo(status, "administer-nodes");
135        Assert.assertNotNull(administrationMojo);
136        administrationMojo.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
137        simulatedLog.reset();
138        administrationMojo.setLog(simulatedLog);
139        administrationMojo.execute();
140        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
141        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
142        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Started"));
143
144        LOGGER.info("   Display nodes2");
145        administrationMojo = (AdministrationMojo) lookupConfiguredMojo(status2, "administer-nodes");
146        Assert.assertNotNull(administrationMojo);
147        administrationMojo.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
148        simulatedLog.reset();
149        administrationMojo.setLog(simulatedLog);
150        administrationMojo.execute();
151        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
152        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
153        assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Started"));
154
155        LOGGER.info("   Browse services");
156        administrationMojo = (AdministrationMojo) lookupConfiguredMojo(status3, "administer-nodes");
157        Assert.assertNotNull(administrationMojo);
158        administrationMojo.environment = new String[]{"BUILD_ID=" + System.getenv("BUILD_ID")};
159        simulatedLog.reset();
160        administrationMojo.setLog(simulatedLog);
161        administrationMojo.execute();
162        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
163        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
164    }
165}
