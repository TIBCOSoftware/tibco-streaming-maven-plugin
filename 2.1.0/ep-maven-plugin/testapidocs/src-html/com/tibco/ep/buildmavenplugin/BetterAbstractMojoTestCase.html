033import java.util.Arrays;
034import java.util.Properties;

036import org.apache.maven.DefaultMaven;
037import org.apache.maven.Maven;
038import org.apache.maven.execution.DefaultMavenExecutionRequest;
039import org.apache.maven.execution.DefaultMavenExecutionResult;
040import org.apache.maven.execution.MavenExecutionRequest;
041import org.apache.maven.execution.MavenExecutionRequestPopulator;
042import org.apache.maven.execution.MavenExecutionResult;
043import org.apache.maven.execution.MavenSession;
044import org.apache.maven.plugin.Mojo;
045import org.apache.maven.plugin.testing.AbstractMojoTestCase;
046import org.apache.maven.project.MavenProject;
047import org.apache.maven.project.ProjectBuilder;
048import org.apache.maven.project.ProjectBuildingRequest;
049import org.eclipse.aether.DefaultRepositorySystemSession;
050import org.eclipse.aether.internal.impl.SimpleLocalRepositoryManagerFactory;
051import org.eclipse.aether.repository.LocalRepository;

053/** Use this as you would {@link AbstractMojoTestCase},
054 * where you want more of the standard maven defaults to be set
055 * (and where the {@link AbstractMojoTestCase} leaves them as null or empty).
056 * This includes:
057 * <ul>
058 * <li> local repo, repo sessions and managers configured
059 * <li> maven default remote repos installed (NB: this does not use your ~/.m2 local settings)
060 * <li> system properties are copies
061 * </ul>
062 * 
063 */ 
064public abstract class BetterAbstractMojoTestCase extends AbstractMojoTestCase {
065
066    File regressionRepository;
067    
068    protected MavenSession newMavenSession() {
069        try {
070            MavenExecutionRequest request = new DefaultMavenExecutionRequest();
071            MavenExecutionResult result = new DefaultMavenExecutionResult();
072
073            regressionRepository = new File(System.getenv("TIBCO_EP_HOME")+"/BUILD/repository");
074            if (regressionRepository.exists()) {
075                request.setLocalRepositoryPath(regressionRepository);
076            } else {
077                regressionRepository = new File(System.getenv("TIBCO_EP_HOME")+"/.repository");
078                if (regressionRepository.exists()) {
079                    request.setLocalRepositoryPath(regressionRepository);
080                }
081            }
082            
083            // populate sensible defaults, including repository basedir and remote repos
084            MavenExecutionRequestPopulator populator;
085            populator = getContainer().lookup( MavenExecutionRequestPopulator.class );
086            populator.populateDefaults( request );
087
088            // this is needed to allow java profiles to get resolved; i.e. avoid during project builds:
089            // [ERROR] Failed to determine Java version for profile java-1.5-detected @ org.apache.commons:commons-parent:22, /Users/alex/.m2/repository/org/apache/commons/commons-parent/22/commons-parent-22.pom, line 909, column 14
090            Properties properties = System.getProperties();
091            properties.put("env.TIBCO_EP_HOME", System.getenv("TIBCO_EP_HOME"));
092            request.setSystemProperties( properties );
093            
094            // and this is needed so that the repo session in the maven session 
095            // has a repo manager, and it points at the local repo
096            // (cf MavenRepositorySystemUtils.newSession() which is what is otherwise done)
097            DefaultMaven maven = (DefaultMaven) getContainer().lookup( Maven.class );
098            DefaultRepositorySystemSession repoSession =
099                (DefaultRepositorySystemSession) maven.newRepositorySession( request );
100            repoSession.setLocalRepositoryManager(
101                new SimpleLocalRepositoryManagerFactory().newInstance(repoSession, 
102                    new LocalRepository( request.getLocalRepository().getBasedir() ) ));
103
104            @SuppressWarnings("deprecation")
105            MavenSession session = new MavenSession( getContainer(), 
106                repoSession,
107                request, result );
108            return session;
109        } catch (Exception e) {
110            throw new RuntimeException(e);
111        }
112    }
113    
114    /** Extends the super to use the new {@link #newMavenSession()} introduced here 
115     * which sets the defaults one expects from maven; the standard test case leaves a lot of things blank */
116    @Override
117    protected MavenSession newMavenSession(MavenProject project) {
118        MavenSession session = newMavenSession();
119        session.setCurrentProject( project );
120        session.setProjects( Arrays.asList( project ) );
121        return session;        
122    }
123
124    /** As {@link #lookupConfiguredMojo(MavenProject, String)} but taking the pom file 
125     * and creating the {@link MavenProject}. 
126     * @param pom pom file
127     * @param goal goal
128     * @return mojo
129     * @throws Exception on error
130     * */
131    protected Mojo lookupConfiguredMojo(File pom, String goal) throws Exception {
132        assertNotNull( pom );
133        assertTrue( pom.exists() );
134
135        ProjectBuildingRequest buildingRequest = newMavenSession().getProjectBuildingRequest();
136        ProjectBuilder projectBuilder = lookup(ProjectBuilder.class);
137        MavenProject project = projectBuilder.build(pom, buildingRequest).getProject();
138
139        return lookupConfiguredMojo(project, goal);
140    }
