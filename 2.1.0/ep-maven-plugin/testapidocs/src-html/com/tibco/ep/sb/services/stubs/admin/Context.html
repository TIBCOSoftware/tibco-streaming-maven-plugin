031package com.tibco.ep.sb.services.stubs.admin;
032
033import com.tibco.ep.sb.services.management.IContext;
034
035import java.nio.file.Path;
036import java.util.HashMap;
037import java.util.Map;
038
039/**
040 * The context stub implementation
041 */
042public class Context extends ServiceAwareStub implements IContext {
043
044    private final Map<String, String> env = new HashMap<>();
045
046    /**
047     * @param adminService The admin service
048     * @param installation The installation path
049     */
050    public Context(RuntimeAdminService adminService, Path installation) {
051        super(adminService);
052        logMethod("constructor", installation);
053    }
054
055    @Override
056    public IContext withTracingEnabled() {
057        logMethod("enableTracing");
058        return this;
059    }
060
061    @Override
062    public IContext withEnvironment(Map<String, String> environment) {
063        logMethod("setEnvironment", environment);
064
065        env.clear();
066        env.putAll(environment);
067
068        return this;
069    }
070
071    @Override
072    public void clearEnvironment() {
073        logMethod("clearEnvironment");
074        env.clear();
075    }
076
077    @Override
078    public String toString() {
079        return "Context{" +
080            "env=" + env +
081            '}';
082    }
083}
